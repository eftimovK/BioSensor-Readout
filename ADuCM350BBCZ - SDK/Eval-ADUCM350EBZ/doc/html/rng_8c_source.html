120 ADI_RNG_RESULT_TYPE
121 adi_RNG_Init(ADI_RNG_DEV_ID_TYPE const InstanceNum, ADI_RNG_HANDLE* const phDevice)
122 {
123  ADI_RNG_HANDLE hDevice;
124 
125  /* store a bad handle in case of failure */
126  *phDevice = (ADI_RNG_HANDLE) NULL;
127 
128 #ifdef ADI_DEBUG
129  if (InstanceNum >= ADI_RNG_MAX_DEVID)
130  {
131  return ADI_RNG_ERR_BAD_INSTANCE;
132  }
133 #endif
134 
135  /* local pointer to instance data */
136  hDevice = &gRNG_DevData[InstanceNum];
137 
138 #ifdef ADI_DEBUG
139  if (hDevice->bInitialized)
140  {
141  return ADI_RNG_ERR_ALREADY_INITIALIZED;
142  }
143 #endif
144 
145  /* first thing: make sure RNG has a clock so it can respond */
146  SystemClockSourcePowerUp (ADI_SYS_CLOCK_SOURCE_LFXTAL, ADI_SYS_CLOCK_TIMEOUT_USE_DEFAULT);
147 
148  /* initialize internal device data */
149  hDevice->pRng = pADI_RNG;
150  hDevice->IRQn = RAND_IRQn;
151  hDevice->cbFunc = NULL;
152  hDevice->cbWatch = (ADI_RNG_INTERRUPT_TYPE) 0;
153 
154  /* clear any pending interrupts */
155  hDevice->pRng->RNGSTAT = ADI_RNG_ALL_INTERRUPTS;
156 
157  /* assert statically configured initialization data */
158  ADI_RNG_CFG_TYPE const* pInitData = &gRngStaticConfigData[InstanceNum];
159 
160  hDevice->pRng->RNGCTL = pInitData->RNGCTL;
161  hDevice->pRng->RNGLEN = pInitData->RNGLEN;
162 
163  /* install interrupt handler (see abstraction macro for OS in adi_rtos.h) */
164  ADI_INSTALL_HANDLER(hDevice->IRQn, Random_Int_Handler);
165 
166  /* mark driver initialized */
167  hDevice->bInitialized = true;
168 
169  /* store handle at application handle pointer */
170  *phDevice = hDevice;
171 
172  return ADI_RNG_SUCCESS; // initialized
173 }

606 ADI_RNG_RESULT_TYPE
607 adi_RNG_GetOscCount (ADI_RNG_HANDLE const hDevice, uint32_t* const pOscCount)
608 {
609  uint32_t tmp;
610 
611 #ifdef ADI_DEBUG
612  if (ADI_RNG_INVALID_HANDLE(hDevice))
613  return ADI_RNG_ERR_BAD_DEV_HANDLE;
614 
615  if (!hDevice->bInitialized)
616  return ADI_RNG_ERR_NOT_INITIALIZED;
617 #endif
618 
619  /* data valid when RNGRDY and CNTEN bits are set */
620  if ((hDevice->pRng->RNGSTAT & BITM_RNG_RNGSTAT_RNGRDY) &&
621  (hDevice->pRng->RNGCTL & BITM_RNG_RNGCTL_CNTEN))
622  {
623  /* Get the oscillator count high count */
624  tmp = hDevice->pRng->RNGCNTH;
625  /* Append the low count and return it */
626  *pOscCount = ((tmp << 16) | hDevice->pRng->RNGCNTL);
627  }
628  else
629  {
630  /* data not valid */
631  *pOscCount = 0;
632 
633  return ADI_RNG_ERR_NOT_VALID;
634  }
635 
636  return ADI_RNG_SUCCESS;
637 }

724 {
725 
726 #ifdef ADI_DEBUG

729 
730  if (!hDevice->bInitialized)
731  return ADI_RNG_ERR_NOT_INITIALIZED;

733  /* if registering a callback */

743  hDevice->cbWatch = cbWatch;
744 
745  /* enable RNG interrupts in NVIC */
746  NVIC_EnableIRQ(hDevice->IRQn);
747 
748  return ADI_RNG_SUCCESS;

753 ADI_INT_HANDLER (Random_Int_Handler)
754 {
755  ADI_RNG_HANDLE pDev = &gRNG_DevData[ADI_RNG_DEVID_0]; // so far, there is only one RNG, so this is safe
756  register uint16_t candidate;
757  register ADI_RNG_INTERRUPT_TYPE fired;
758 
759  /* if we have an initialized driver and a registered callback... */
760  if (pDev->bInitialized && pDev->cbFunc) {
761 
762  /* determin candidate interrupt source(s) */
763 ADI_ENTER_CRITICAL_REGION();
764  /* read status register without other interrupts in between */
765  candidate = (pDev->pRng->RNGSTAT & ADI_RNG_ALL_INTERRUPTS);
766 ADI_EXIT_CRITICAL_REGION();
767 
768  fired = (ADI_RNG_INTERRUPT_TYPE)0;
769 
770  /* need to test each interrupt source and whether it is enabled before notifying */
771  /* because each source is latched regardless of whether it is enabled or not :-( */
772  /* eventually, 'fired' now represents interrupts that are both set and enabled... */
773  if (ADI_RNG_INTERRUPT_RNG_RDY & candidate)
774  fired |= ADI_RNG_INTERRUPT_RNG_RDY;
775 
776  /* forward the interrupt to the user if he is watching it and it has fired */
777  /* pass the "fired" value as the event. argument param is not used */
778  if (pDev->cbWatch & fired)
779  pDev->cbFunc (pDev, fired, NULL);
780 
781  /* clear the watched interrupt(s) that fired */
782  pDev->pRng->RNGSTAT = fired;
783  }
784 }
