374 ADI_SPI_RESULT_TYPE adi_SPI_MasterInit (ADI_SPI_DEV_ID_TYPE const devID,
375  ADI_SPI_DEV_HANDLE* const phDevice)
376 {
377  ADI_SPI_DEV_HANDLE hDevice = &SPI_DevData[devID];
378  ADI_SPI_RESULT_TYPE result = ADI_SPI_SUCCESS;
379 
380  /* store a bad handle in case of failure */
381  *phDevice = (ADI_SPI_DEV_HANDLE) NULL;
382 
383 #ifdef ADI_DEBUG
384  if (devID >= ADI_SPI_MAX_DEVID)
385  return ADI_SPI_ERR_BAD_DEV_ID;
386 
387  if (ADI_SPI_STATE_UNINITIALIZED != hDevice->InitState)
388  return ADI_SPI_ERR_ALREADY_INITIALIZED;
389 #endif
390 
391  /* initialize GPIO */
392  if (ADI_GPIO_SUCCESS != adi_GPIO_Init()) {
393  hDevice->InitState = ADI_SPI_STATE_UNINITIALIZED;
394  return ADI_SPI_ERR_BAD_GPIO_CONFIG;
395  }
396 
397  result = CommonInit( hDevice, devID);
398  if( result != ADI_SPI_SUCCESS )
399  {
400  return result;
401  }
402 
403 #if (ADI_SPI_CFG_ENABLE_STATIC_CONFIG_SUPPORT == 1)
404  {
405  ADI_SPI_CFG_TYPE const* pSPICfg = &gSPICfg[devID];
406 
407  /* Initialize the device based on the given configuration parameters */
408  hDevice->pSpi->SPICON = pSPICfg->SPI_CFG;
409  hDevice->pSpi->SPIDIV = pSPICfg->SPI_DIV;
410 
411  }
412 #endif /* ADI_SPI_CFG_ENABLE_STATIC_CONFIG_SUPPORT */
413 
414 
415  /* mark device initialization in progress */
416  /* "ADI_SPI_STATE_INITIALIZING" allows use of common set bitrate */
417  /* function (during "ConfigureCoreAsMaster") function to pass without */
418  /* failing on "ADI_SPI_STATE_UNINITIALIZED" */
419  hDevice->InitState = ADI_SPI_STATE_INITIALIZING;
420 
421  /* configure the default CS port/pin, as already designated by CommonInit() */
422  if (ADI_SPI_SUCCESS != ConfigureMasterCS (hDevice, true)) {
423  hDevice->InitState = ADI_SPI_STATE_UNINITIALIZED;
424  return ADI_SPI_ERR_BAD_GPIO_CONFIG;
425  }
426 
427  /* configure core for Master-Mode operation */
428  if (ADI_SPI_SUCCESS != (result = ConfigureCoreAsMaster (hDevice))) {
429  hDevice->InitState = ADI_SPI_STATE_UNINITIALIZED;
430  return result;
431  }
432 
433  /* enable the dedicated SPI core clock */
434  if (pSPI0_DevData == hDevice) SystemEnableClock(ADI_SYS_CLOCK_GATE_SPI0, true);
435  if (pSPI1_DevData == hDevice) SystemEnableClock(ADI_SYS_CLOCK_GATE_SPI1, true);
436  if (pSPIH_DevData == hDevice) SystemEnableClock(ADI_SYS_CLOCK_GATE_SPIH, true);
437 
438  /* mark device fully initialized */
439  hDevice->InitState = ADI_SPI_STATE_MASTER;
440 
441  /* write the device data pointer into the caller's handle */
442  *phDevice = hDevice;
443 
444  return result;
445 }

504  }
505

510  hDevice->InitState = ADI_SPI_STATE_INITIALIZING;
511 
512  /* configure core for Slave-Mode operation */
513  if (ADI_SPI_SUCCESS != (result = ConfigureCoreAsSlave (hDevice))) {
514  hDevice->InitState = ADI_SPI_STATE_UNINITIALIZED;

517

519  ADI_ENABLE_INT(hDevice->IRQn);
520

522  if (pSPI0_DevData == hDevice) SystemEnableClock(ADI_SYS_CLOCK_GATE_SPI0, true);
523  if (pSPI1_DevData == hDevice) SystemEnableClock(ADI_SYS_CLOCK_GATE_SPI1, true);
524  if (pSPIH_DevData == hDevice) SystemEnableClock(ADI_SYS_CLOCK_GATE_SPIH, true);
525 
526  /* mark device fully initialized */
527  hDevice->InitState = ADI_SPI_STATE_SLAVE;
528 
529  /* write the device data pointer into the caller's handle */
530  *phDevice = hDevice;
531 
532  return result;
533 }

553 ADI_SPI_RESULT_TYPE adi_SPI_UnInit (ADI_SPI_DEV_HANDLE const hDevice)
554 {
555 
556  ADI_SPI_RESULT_TYPE result = ADI_SPI_SUCCESS;
557 
558 #ifdef ADI_DEBUG
559  if (ADI_SPI_INVALID_HANDLE(hDevice))
560  return ADI_SPI_ERR_BAD_DEV_HANDLE;
561 
562  if ((hDevice->InitState != ADI_SPI_STATE_MASTER) && (hDevice->InitState != ADI_SPI_STATE_SLAVE))
563  return ADI_SPI_ERR_NOT_INITIALIZED ;
564 #endif
565 
566  /* disable Interrupt */
567  ADI_DISABLE_INT(hDevice->IRQn);
568  ADI_UNINSTALL_HANDLER(hDevice->IRQn);
569 
570 #if (ADI_CFG_ENABLE_RTOS_SUPPORT == 1)
571  /* destroy semaphore */
572  if (adi_osal_SemDestroyStatic(hDevice->hSem) != ADI_OSAL_SUCCESS)
573  {
574  return ADI_SPI_ERR_SEMAPHORE_FAILED;
575  }
576 #endif /* ADI_CFG_ENABLE_RTOS_SUPPORT */
577 
578  /* uninit internals */
579  if (ADI_SPI_SUCCESS != hDevice->FunctionTable.pUnInitFcn(hDevice))
580  return ADI_SPI_ERR_BAD_DMA_CONFIG;
581 
582  /* Uninitialize the GPIO driver */
583  if (ADI_GPIO_SUCCESS != adi_GPIO_UnInit())
584  return ADI_SPI_ERR_BAD_GPIO_CONFIG;
585 
586  /* disable the SPI controller */
587  hDevice->pSpi->SPICON = 0;
588 
589  /* disable the dedicated SPI core clock */
590  if (pSPI0_DevData == hDevice) SystemEnableClock(ADI_SYS_CLOCK_GATE_SPI0, false);
591  if (pSPI1_DevData == hDevice) SystemEnableClock(ADI_SYS_CLOCK_GATE_SPI1, false);
592  if (pSPIH_DevData == hDevice) SystemEnableClock(ADI_SYS_CLOCK_GATE_SPIH, false);
593 
594  /* invalidate initialization state */
595  hDevice->InitState = ADI_SPI_STATE_UNINITIALIZED;
596 
597  return result;
598 }
