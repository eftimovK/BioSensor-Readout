65 ADI_METRIC_RESULT_TYPE adi_Metric_Init(ADI_GPT_DEV_ID_TYPE InstanceNum, ADI_METRIC_DEV_HANDLE* const pHandle)
66 {
67  ADI_METRIC_DEV_HANDLE hDevice = pMetric_DevData;
68  ADI_METRIC_RESULT_TYPE result = ADI_METRIC_SUCCESS;
69 
70  *pHandle = hDevice;
71 
72  /* Timer 0 */
73  hDevice->OverflowIndex = 0;
74  hDevice->TestIndex = 0;
75  hDevice->Count = 0;
76  hDevice->Accumulate = 0;
77 
78  /* open timer instance */
79  if(adi_GPT_Init(InstanceNum, &hDevice->hGpTimer)!= ADI_GPT_SUCCESS)
80  {
81  return ADI_METRIC_ERROR;
82  }
83  if(adi_GPT_SetClockSelect(hDevice->hGpTimer, ADI_GPT_CLOCK_SELECT_HFOSC)!= ADI_GPT_SUCCESS)
84  {
85  return ADI_METRIC_ERROR;
86  }
87  if(adi_GPT_SetCountMode(hDevice->hGpTimer, ADI_GPT_COUNT_UP)!= ADI_GPT_SUCCESS)
88  {
89  return ADI_METRIC_ERROR;
90  }
91  if(adi_GPT_SetPeriodicMode(hDevice->hGpTimer, false, 0)!= ADI_GPT_SUCCESS)
92  {
93  return ADI_METRIC_ERROR;
94  }
95  if(adi_GPT_SetLdVal(hDevice->hGpTimer, 0)!= ADI_GPT_SUCCESS)
96  {
97  return ADI_METRIC_ERROR;
98  }
99  if(adi_GPT_SetPrescaler(hDevice->hGpTimer, ADI_GPT_PRESCALER_16)!= ADI_GPT_SUCCESS)
100  {
101  return ADI_METRIC_ERROR;
102  }
103  if(adi_GPT_RegisterCallback(hDevice->hGpTimer, GPTimerCallback, hDevice->hGpTimer)!= ADI_GPT_SUCCESS)
104  {
105  return ADI_METRIC_ERROR;
106  }
107 
108  return result;
109 }

148 ADI_METRIC_RESULT_TYPE adi_Metric_Start(ADI_METRIC_DEV_HANDLE const hDevice)
149 {
150  ADI_METRIC_RESULT_TYPE result = ADI_METRIC_SUCCESS;
151 #if defined (USE_ENABLE_API)
152  /* this API call can consume considerable cycle counts when RTOS is enabled */
153  if(adi_GPT_SetTimerEnable(hDevice->hGpTimer, true) != ADI_GPT_SUCCESS)
154  {
155  return ADI_METRIC_ERROR;
156  }
157 #else
158  /* bypass timer API call to avoid consuming cycle counts */
159  pADI_GPT0->GPTCON |= TCON_ENABLE;
160 #endif
161  return result;
162 }

177 ADI_METRIC_RESULT_TYPE adi_Metric_Stop(ADI_METRIC_DEV_HANDLE const hDevice)
178 {
179  ADI_METRIC_RESULT_TYPE result = ADI_METRIC_SUCCESS;
180 #if defined (USE_ENABLE_API)
181  if(adi_GPT_SetTimerEnable(hDevice->hGpTimer, false) != ADI_GPT_SUCCESS)
182  {
183  return ADI_METRIC_ERROR;
184  }
185 #else
186  pADI_GPT0->GPTCON &= ~TCON_ENABLE;
187 #endif
188  hDevice->Accumulate += ((hDevice->OverflowIndex * 0xFFFF) + hDevice->Count);
189  hDevice->OverflowIndex = 0;
190  hDevice->Count = 0;
191 
192  return result;
193 }
