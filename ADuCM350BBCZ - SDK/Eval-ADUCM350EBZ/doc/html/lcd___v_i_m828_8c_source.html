293 ADI_LCDVIM828_RESULT_TYPE adi_LCDVIM828_DisplayString(ADI_LCD_DEV_HANDLE const hDevice,
294  ADI_LCD_SCREEN_TYPE const Screen,
295  uint8_t const DisplayString[])
296 {
297  int32_t i;
298  int32_t j;
299  int32_t numDp = 0;
300  uint8_t tempDisplayString[ADI_LCDVIM828_NUM_SEGMENTS];
301  uint8_t charDP[] = {0,0,0,0,0,0,0,0}; /* Records the position of a decimal point */
302  bool_t addDP;
303  uint32_t charOffset; /* Holds the offset of character in the mapping table */
304 
305 #ifdef ADI_DEBUG
306  if(DisplayString == NULL)
307  {
308  return ADI_LCDVIM828_ERR_NULL_POINTER;
309  }
310 #endif /* ADI_DEBUG */
311 
312  /* Use a temporary variable so that the original string is not modified */
313  for (i = 0; i < ADI_LCDVIM828_NUM_SEGMENTS; i++)
314  {
315  tempDisplayString[i] = DisplayString[i];
316 
317  if(tempDisplayString[i] == '.')
318  {
319  /* count the decimal points */
320  numDp++;
321  }
322  }
323 
324  /* Clear the measurement display first */
325  adi_LCDVIM828_ClearDisplay(hDevice, Screen);
326 
327  charOffset = LCDVIM828_SPACE_ASCII_TO_INDEX;
328 
329  /* Handle decimal point(s) display
330  * cases:
331  * - numeric value with one decimal point (0.1)
332  * - decimal point in first position (.1)
333  * - multiple decimal points (0..1)
334  */
335  if ((numDp == 1) && (tempDisplayString[0] == '.'))
336  {
337  /* decimal in first position */
338  tempDisplayString[0] = ' ';
339  charDP[0] = 1;
340  }
341  else if (numDp == 1)
342  {
343  /* one decimal point not in first position (assume numeric value)
344  * shift the display characters after the decimal
345  */
346  for (i = 1; i < ADI_LCDVIM828_NUM_SEGMENTS; i++)
347  {
348  if (tempDisplayString[i] == '.')
349  {
350  /* record the position and shift */
351  charDP[i-1] = 1;
352 
353  for (j = i; j < ADI_LCDVIM828_NUM_SEGMENTS; j++)
354  {
355  if (tempDisplayString[j + 1] == '\0')
356  {
357  tempDisplayString[j] = ' ';
358  }
359  else
360  {
361  tempDisplayString[j] = tempDisplayString[j + 1];
362  }
363  }
364  break;
365  }
366  }
367  }
368  else if (numDp > 1)
369  {
370  /* multiple decimal points - display as is */
371  for (i = 0; i < ADI_LCDVIM828_NUM_SEGMENTS; i++)
372  {
373  if (tempDisplayString[i] == '.')
374  {
375  /* record the position */
376  charDP[i] = 1;
377  tempDisplayString[i] = ' ';
378  }
379  }
380  }
381 
382  /* Display the given string */
383  for(i = 0; i < ADI_LCDVIM828_NUM_SEGMENTS; i++)
384  {
385  /* Check if a decimal point is used */
386  if(charDP[i])
387  {
388  addDP = true;
389  }
390  /* Check if character is valid and assign offset */
391  if ((tempDisplayString[i] >= '0') && (tempDisplayString[i] <= '9'))
392  {
393  charOffset = LCDVIM828_ASCII_NUMBER_TO_INDEX;
394  }
395  else if((tempDisplayString[i] >= 'A') && (tempDisplayString[i] <= 'Z'))
396  {
397  charOffset = LCDVIM828_UPPERCASE_ASCII_TO_INDEX;
398  }
399  else if(tempDisplayString[i]=='-')
400  {
401  charOffset = LCDVIM828_MINUS_ASCII_TO_INDEX;
402  }
403  else if(tempDisplayString[i]==' ')
404  {
405  charOffset = LCDVIM828_SPACE_ASCII_TO_INDEX;
406  }
407  else
408  {
409  return ADI_LCDVIM828_ERR_BAD_CHAR;
410  }
411 
412  DisplayChar(hDevice, Screen, &SegmentRegInfo[i], (tempDisplayString[i] - charOffset), addDP);
413 
414  addDP = false;
415  }
416 
417  return ADI_LCDVIM828_SUCCESS;
418 }

443 /* Sets the LCD registers to clear the characters */
444 static ADI_LCDVIM828_RESULT_TYPE ClearChars(ADI_LCD_DEV_HANDLE const hDevice,
445  ADI_LCD_SCREEN_TYPE const Screen,
446  ADI_LCDVIM828_REG_INFO const *pRegInfoTable,
447  uint8_t const NumChars)
448 {
449  ADI_LCD_DATA_TYPE DataTable[2];
450  uint8_t i;
451  ADI_LCDVIM828_REG_INFO const *pRegInfo;
452 
453  for(i = 0; i < NumChars; i++)
454  {
455  pRegInfo = &pRegInfoTable[i];
456 
457  /* Fill the register table whose contents to be read */
458  DataTable[0].LCDRegNumber = pRegInfo->RegNum1;
459  DataTable[1].LCDRegNumber = pRegInfo->RegNum2;
460 
461  /* Read the register contents */
462  adi_LCD_GetDataReg(hDevice, Screen, DataTable, 2);
463 
464  /* Set the new data */
465  DataTable[0].RegValue &= 0;
466  DataTable[1].RegValue &= 0;
467 
468  /* Set back the register contents */
469  adi_LCD_SetDataReg(hDevice, Screen, DataTable, 2);
470  }
471 
472  return ADI_LCDVIM828_SUCCESS;
473 }

475 /* Sets the LCD registers for the given character */
476 static ADI_LCDVIM828_RESULT_TYPE DisplayChar(ADI_LCD_DEV_HANDLE const hDevice,
477  ADI_LCD_SCREEN_TYPE const Screen,
478  ADI_LCDVIM828_REG_INFO const *pRegInfo,
479  uint32_t CharIndex,
480  bool_t dpReqd)
481 {
482  ADI_LCD_DATA_TYPE DataTable[2];
483  uint16_t MapValue = pRegInfo->pMapTable[CharIndex];
484  uint16_t temp1;
485  uint16_t temp2;
486 
487  /* If a decimal point is required, amend the map value */
488  if (dpReqd)
489  {
490  MapValue |= pRegInfo->pMapTable[LCDVIM828_DP_OFFSET];
491  }
492 
493  /* Fill the register table whose contents to be read */
494  DataTable[0].LCDRegNumber = pRegInfo->RegNum1;
495  DataTable[1].LCDRegNumber = pRegInfo->RegNum2;
496 
497  /* Read the register contents */
498  adi_LCD_GetDataReg(hDevice, Screen, DataTable, 2);
499 
500  temp1 = DataTable[0].RegValue;
501  temp2 = DataTable[1].RegValue;
502 
503  /* Set the new data */
504  DataTable[0].RegValue = temp1 | LCD_VIM828_CHAR_REG1_VALUE(MapValue, pRegInfo->BitPos1);
505  DataTable[1].RegValue = temp2 | LCD_VIM828_CHAR_REG1_VALUE(MapValue, pRegInfo->BitPos2);
506 
507  /* Set back the register contents */
508  adi_LCD_SetDataReg(hDevice, Screen, DataTable, 2);
509 
510  return ADI_LCDVIM828_SUCCESS;
511 
512  }
