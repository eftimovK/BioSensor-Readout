137 ADI_PDI_RESULT_TYPE adi_PDI_Init (ADI_PDI_DEV_ID_TYPE const devID, ADI_PDI_DEV_HANDLE* const phDevice)
138 {
139  ADI_PDI_DEV_HANDLE hDevice = &PDI_DevData[devID];
140 
141  /* store a bad handle in case of failure */
142  *phDevice = (ADI_PDI_DEV_HANDLE) NULL;
143 
144 #ifdef ADI_DEBUG
145  if (devID >= ADI_PDI_MAX_DEVID)
146  return ADI_PDI_ERR_BAD_DEV_ID;
147 
148  if (ADI_PDI_STATE_UNINITIALIZED != hDevice->InitState)
149  return ADI_PDI_ERR_ALREADY_INITIALIZED;
150 #endif
151 
152  hDevice->pPdi = pADI_PDI;
153 
154  /* assert statically configured initialization data, if present,
155  in lieu of dynamic initializations */
156 
157  ADI_PDI_STATIC_INIT_TYPE const* pInitData = &gPdiStaticInitData[devID];
158 
159  hDevice->pPdi->PDI_INT_SET = pInitData->pdiInterruptEnable;
160  hDevice->pPdi->PDI_INT_CLR = pInitData->pdiInterruptDisable;
161  hDevice->pPdi->PDI_STAT = pInitData->pdiInterruptClear;
162  hDevice->pPdi->PDI_IF_TIMING = pInitData->pdiTimingControl;
163 
164  /* mark device fully initialized */
165  hDevice->InitState = ADI_PDI_STATE_INITIALIZED;
166 
167  /* write the device data pointer into the caller's handle */
168  *phDevice = hDevice;
169 
170  return ADI_PDI_SUCCESS;
171 }

430 ADI_PDI_RESULT_TYPE adi_PDI_SetCommand (ADI_PDI_DEV_HANDLE const hDevice,
431  const uint8_t ParamCount,
432  const ADI_PDI_CMD_PARAM_TYPE Type,
433  const ADI_PDI_CMD_DIR_TYPE Dir,
434  const uint8_t Command)
435 {
436  uint32_t pdiCmd;
437 
438 #ifdef ADI_DEBUG
439  if (ADI_PDI_INVALID_HANDLE(hDevice))
440  return ADI_PDI_ERR_BAD_DEV_HANDLE;
441 
442  if (hDevice->InitState != ADI_PDI_STATE_INITIALIZED)
443  return ADI_PDI_ERR_NOT_INITIALIZED ;
444 #endif
445 
446  /* build the command locally */
447  pdiCmd = Command | (ParamCount << ADI_PDI_CMD_PARAM_COUNT_POS);
448 
449  /* set the frame data bit if its frame data */
450  if (PDI_CMD_DATA == Type)
451  pdiCmd |= ADI_PDI_CMD_FRAME_DATA;
452 
453  /* set the write bit if its a write */
454  if (PDI_DIR_WRITE == Dir)
455  pdiCmd |= ADI_PDI_CMD_DIRECTION_WRITE;
456 
457  /* clear CMD_DONE status prior to command write */
458  /* don't clear CMD_WR_ERR... let it accumulate in case of parameter count mismatch */
459  pADI_PDI->PDI_STAT = ADI_PDI_STATUS_CMD_DONE;
460 
461  /* write the command register */
462  pADI_PDI->PDI_CMD = pdiCmd;
463 
464  /* the required command wait is done in the device-specific command *\
465  \* translation switch statement, after each call to this API */
466 
467  return ADI_PDI_SUCCESS;
468 }

737 ADI_PDI_RESULT_TYPE adi_PDI_DispatchMipiCommand (ADI_PDI_DEV_HANDLE const hDevice, const ADI_MIPI_DCS_ENUM MipiCmd, const void* pMipiParamBlock)
738 {
739  ADI_PDI_RESULT_TYPE result = ADI_PDI_SUCCESS;
740 
741 #ifdef ADI_DEBUG
742  if (ADI_PDI_INVALID_HANDLE(hDevice))
743  return ADI_PDI_ERR_BAD_DEV_HANDLE;
744 
745  if (hDevice->InitState != ADI_PDI_STATE_INITIALIZED)
746  return ADI_PDI_ERR_NOT_INITIALIZED ;
747 #endif
748 
749  /* pass it down to the native MIPI translation layer */
750  if (ADI_PDI_SUCCESS != (result = adi_TranslateMipiCommand(hDevice, MipiCmd, pMipiParamBlock)))
751  return result;
752 
753  return result;
754 }
