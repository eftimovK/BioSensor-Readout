129 ADI_WUT_RESULT_TYPE adi_WUT_Init (ADI_WUT_DEV_ID_TYPE const devID, ADI_WUT_DEV_HANDLE* const phDevice)
130 {
131  ADI_WUT_DEV_HANDLE hDevice = &WUT_DevData[devID];
132  ADI_WUT_RESULT_TYPE result = ADI_WUT_SUCCESS;
133 
134  /* store a bad handle in case of failure */
135  *phDevice = (ADI_WUT_DEV_HANDLE) NULL;
136 
137 #ifdef ADI_DEBUG
138  if (devID != ADI_WUT_DEVID_0)
139  return ADI_WUT_ERR_BAD_DEV_ID;
140 
141  if (ADI_WUT_STATE_UNINITIALIZED != pWUT0_DevData->InitState)
142  return ADI_WUT_ERR_ALREADY_INITIALIZED;
143 #endif
144 
145  /* common inits */
146  hDevice->InitState = ADI_WUT_STATE_UNINITIALIZED;
147  hDevice->pWut = pADI_WUT;
148 
149 
150  /* assert pre-configured static data, allowing linker to eliminate unused dynamic config APIs */
151  ADI_WUT_STATIC_INIT_TYPE const* pInitData = &gWutStaticInitData[devID];
152 
153  hDevice->pWut->T2CON = pInitData->wdtControlRegister;
154  hDevice->pWut->T2INC = pInitData->wdtIntervalRegister;
155 
156  hDevice->pWut->T2WUFA1 = (uint16_t)((pInitData->wdtCompareARegister & 0xffff0000) >> 16);
157  hDevice->pWut->T2WUFA0 = (uint16_t)((pInitData->wdtCompareARegister & 0x0000ffff) >> 0);
158 
159  hDevice->pWut->T2WUFB1 = (uint16_t)((pInitData->wdtCompareBRegister & 0xffff0000) >> 16);
160  hDevice->pWut->T2WUFB0 = (uint16_t)((pInitData->wdtCompareBRegister & 0x0000ffff) >> 0);
161 
162  hDevice->pWut->T2WUFC1 = (uint16_t)((pInitData->wdtCompareCRegister & 0xffff0000) >> 16);
163  hDevice->pWut->T2WUFC0 = (uint16_t)((pInitData->wdtCompareCRegister & 0x0000ffff) >> 0);
164 
165  hDevice->pWut->T2WUFD1 = (uint16_t)((pInitData->wdtCompareDRegister & 0xffff0000) >> 16);
166  hDevice->pWut->T2WUFD0 = (uint16_t)((pInitData->wdtCompareDRegister & 0x0000ffff) >> 0);
167 
168  hDevice->pWut->T2IEN = pInitData->wdtInterruptEnableRegister;
169  hDevice->pWut->T2CLRI = pInitData->wdtInterruptClearRegister;
170 
171  /* interrupt ID */
172  hDevice->IRQn = WUT_IRQn;
173 
174  /* install interrupt handler and default callback attributes (see abstraction macro for OS in adi_rtos.h) */
175  ADI_INSTALL_HANDLER(hDevice->IRQn, WakeUp_Int_Handler);
176  hDevice->cbFunc = NULL;
177  hDevice->cbWatch = (ADI_WUT_TARGET_TYPE) 0;
178 
179 
180  /* mark device initialized */
181  hDevice->InitState = ADI_WUT_STATE_INITIALIZED;
182 
183  /* write the device data pointer into the caller's handle */
184  *phDevice = hDevice;
185 
186  return result;
187 }

204 ADI_WUT_RESULT_TYPE adi_WUT_UnInit(ADI_WUT_DEV_HANDLE const hDevice)
205 {
206  ADI_WUT_RESULT_TYPE result = ADI_WUT_SUCCESS;
207 
208 #ifdef ADI_DEBUG
209  if (pWUT0_DevData != hDevice)
210  return ADI_WUT_ERR_BAD_DEV_HANDLE;

224  hDevice->cbFunc = NULL;
225  hDevice->cbWatch = (ADI_WUT_TARGET_TYPE)0;
226 
227  return result;
228 }

255 ADI_WUT_RESULT_TYPE adi_WUT_SetClockSelect (ADI_WUT_DEV_HANDLE const hDevice, const ADI_WUT_CLOCK_SOURCE_TYPE ClockSource)
256 {
257  ADI_WUT_RESULT_TYPE result = ADI_WUT_SUCCESS;
258 
259 #ifdef ADI_DEBUG
260  if (pWUT0_DevData != hDevice)
261  return ADI_WUT_ERR_BAD_DEV_HANDLE;
262 
263  if (hDevice->InitState != ADI_WUT_STATE_INITIALIZED)
264  return ADI_WUT_ERR_NOT_INITIALIZED;
265 
266  if ( (ADI_WUT_CLK_PCLK != ClockSource)
267  && (ADI_WUT_CLK_LFXTAL != ClockSource)
268  && (ADI_WUT_CLK_LFOSC != ClockSource)
269  )
270  return ADI_WUT_ERR_PARAM_OUT_OF_RANGE;
271 #endif
272 
273  // set new value
274  ADI_ENTER_CRITICAL_REGION();
275  hDevice->pWut->T2CON &= ~T2CON_CLK_MSK;
276  hDevice->pWut->T2CON |= (ClockSource << 9);
277  ADI_EXIT_CRITICAL_REGION();
278 
279  return result;

333 ADI_WUT_RESULT_TYPE adi_WUT_SetComparator (ADI_WUT_DEV_HANDLE const hDevice,
334  const ADI_WUT_TARGET_TYPE target,
335  const uint32_t value)
336 {
337  ADI_WUT_RESULT_TYPE result = ADI_WUT_SUCCESS;
338  bool_t savedEnable, savedInterrupt;
339  uint16_t upper, lower;
340 
341 #ifdef ADI_DEBUG
342  if (pWUT0_DevData != hDevice)
343  return ADI_WUT_ERR_BAD_DEV_HANDLE;
344 
345  if (hDevice->InitState != ADI_WUT_STATE_INITIALIZED)
346  return ADI_WUT_ERR_NOT_INITIALIZED;
347 
348  /* exclude all but comparators B/C/D */
349  if (~(ADI_WUT_COMPB | ADI_WUT_COMPC | ADI_WUT_COMPD) & target)
350  return ADI_WUT_ERR_PARAM_OUT_OF_RANGE;
351 
352 #endif
353 
354  /* save/disable timer state */
355  if (ADI_WUT_SUCCESS == result)
356  savedEnable = adi_WUT_GetTimerEnable(hDevice);
357  if (ADI_WUT_SUCCESS == result)
358  result = adi_WUT_SetTimerEnable(hDevice, false);
359 
360  /* save/disable interrupt state */
361  if (ADI_WUT_SUCCESS == result)
362  savedInterrupt = adi_WUT_GetInterruptEnable(hDevice, target);
363  if (ADI_WUT_SUCCESS == result)
364  result = adi_WUT_SetInterruptEnable(hDevice, target, false);
365 
366  /* no need to freeze upper/lower comparator writes because the timer is disabled */
367  if (ADI_WUT_SUCCESS == result) {
368 
369  /* break it up */
370  upper = (uint16_t)((value & 0xffff0000) >> 16);
371  lower = (uint16_t) (value & 0x0000ffff);
372 
373  /* set compB */
374  if (ADI_WUT_COMPB & target) {
375  hDevice->pWut->T2WUFB1 = upper;
376  hDevice->pWut->T2WUFB0 = lower;
377  }
378 
379  /* set compC */
380  if (ADI_WUT_COMPC & target) {
381  hDevice->pWut->T2WUFC1 = upper;
382  hDevice->pWut->T2WUFC0 = lower;
383  }
384 
385  /* set compD */
386  if (ADI_WUT_COMPD & target) {
387  hDevice->pWut->T2WUFD1 = upper;
388  hDevice->pWut->T2WUFD0 = lower;
389  }
390  }
391 
392  /* restore state */
393  if (ADI_WUT_SUCCESS == result)
394  result = adi_WUT_SetInterruptEnable(hDevice, target, savedInterrupt);
395  if (ADI_WUT_SUCCESS == result)
396  result = adi_WUT_SetTimerEnable(hDevice, savedEnable);
397 
398  return result;
399 }

856 uint32_t adi_WUT_GetCurrentCount (ADI_WUT_DEV_HANDLE const hDevice)
857 {

872  return result;
