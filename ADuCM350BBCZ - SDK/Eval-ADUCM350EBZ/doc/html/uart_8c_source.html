304 ADI_INT_HANDLER(UART_Int_Handler)
305 {
306  ADI_UART_HANDLE hDevice = &UART_DevData[ADI_UART_DEVID_0];
307  ADI_UART_TypeDef *pUART_Regs = hDevice->pUartRegs;
308  ADI_UART_CIRC_BUFFER_TYPE *pBuf = NULL;
309  uint8_t iir = hDevice->pUartRegs->COMIIR;
310  uint8_t data;
311 
312  /* iir bit zero is 1 then there is no interrupt */
313  if( iir & COMIIR_NINT )
314  return;
315 
316  /* we got an uart interrupt, process it */
317  switch( iir & COMIIR_STA_MSK )
318  {
319  /* rx line status interupt */
320  case COMIIR_STA_RXLINESTATUS:
321  {
322  uint8_t status = pUART_Regs->COMLSR;
323 
324  ADI_UART_STAT_INC(hDevice->Stats.RxStatusIrqCnt);
325 
326  /* Trap overrun, parity, framing errors */
327  if( status & (COMLSR_OE | COMLSR_PE | COMLSR_FE) )
328  UART_RX_Status_Handler(hDevice, status);
329  }
330  break;
331 
332  /* rx buffer full interrupt */
333  case COMIIR_STA_RXBUFFULL:
334  {
335  pBuf = &hDevice->RxBuffer;
336 
337  /* flush data if no space available */
338  if( pBuf->NumAvailable == pBuf->BufSize )
339  {
340  ADI_UART_STAT_INC(hDevice->Stats.RxDroppedByteCnt);
341 
342  /* trap receive buffer overflow condition. This condition occurs when
343  * internal driver buffer is full and more data is received.
344  *
345  * NOTE: If this condition occurs applications has to adjust the rx buffer size that
346  * is passed via adi_UART_Init() call. The below while(1) is expected to be changed
347  * depending on the application requirements.
348  */
349  while(1); /* internal driver buffer overflow */
350  }
351 
352  /* adjust the write index if reaches the end */
353  if( pBuf->WrIndx == pBuf->BufSize )
354  pBuf->WrIndx = 0;
355 
356  /* read data to rx buffer */
357  data = pUART_Regs->COMRX;
358  pBuf->Buffer[pBuf->WrIndx++] = data;
359 
360  /* increment number of available bytes */
361  pBuf->NumAvailable++;
362  ADI_UART_STAT_INC(hDevice->Stats.RxBufferFullIrqCnt);
363  }
364  break;
365 
366  /* tx buffer empty interrupt */
367  case COMIIR_STA_TXBUFEMPTY:
368  {
369  ADI_UART_STAT_INC(hDevice->Stats.TxBufferEmptyIrqCnt);
370 
371  pBuf = &hDevice->TxBuffer;
372 
373  if( pBuf->RdIndx >= pBuf->BufSize )
374  pBuf->RdIndx = 0;
375 
376  if(pBuf->NumAvailable < pBuf->BufSize)
377  {
378  hDevice->pUartRegs->COMTX = pBuf->Buffer[pBuf->RdIndx++];
379  pBuf->NumAvailable++;
380  }
381  else
382  {
383  /* no more data to transmit so disable tx empty interrupt */
384  pUART_Regs->COMIEN &= ~(COMIEN_ETBEI);
385  }
386  }
387  break;
388 
389  /* modem status interrupt */
390  case COMIIR_STA_MODEMSTATUS:
391  {
392  ADI_UART_STAT_INC(hDevice->Stats.ModemStatusIrqCnt);
393  }
394  break;
395 
396  default:
397  break;
398  }
399 
400 #if (1 == ADI_CFG_ENABLE_RTOS_SUPPORT)
401  adi_osal_SemPost(hDevice->hSem);
402 #else /* (0 == ADI_CFG_ENABLE_RTOS_SUPPORT) */
403  /* Any interrupt that changes the UART device state needs to bring the
404  * processor out of low-power mode, as functions may be waiting for the
405  * state to change.
406  */
407  SystemExitLowPowerMode(&hDevice->bInterruptFlag);
408 #endif /* (0 == ADI_CFG_ENABLE_RTOS_SUPPORT) */
409 }
410 #endif /* (1 == ADI_UART_CFG_INTERRUPT_MODE_SUPPORT) */

424 static ADI_UART_RESULT_TYPE do_read(ADI_UART_HANDLE const hDevice, uint8_t *pRxData, const int16_t Size)
425 {
426  int16_t i = 0;
427 
428 #if (1 == ADI_UART_CFG_INTERRUPT_MODE_SUPPORT)
429  if(IS_INTERRUPT_MODE(hDevice))
430  {
431  int16_t BytesToCopy = 0;
432  int16_t BytesRemaining= 0;
433  ADI_UART_CIRC_BUFFER_TYPE *pBuf = &hDevice->RxBuffer;
434 
435  /* get number of bytes till end */
436  BytesRemaining = pBuf->BufSize - pBuf->RdIndx;
437 
438  /* interrupt mode */
439  BytesToCopy = (BytesRemaining > Size) ? Size : BytesRemaining;
440 
441  /* TODO: check if memcpy is faster */
442  for(i=0;i < BytesToCopy; i++)
443  *pRxData++ = pBuf->Buffer[pBuf->RdIndx++];
444 
445  /* check if we have more to copy */
446  if(BytesToCopy < Size)
447  {
448  BytesRemaining = Size - BytesToCopy;
449  pBuf->RdIndx = 0;
450 
451  for(i=0;i<BytesRemaining;i++)
452  *pRxData++ = pBuf->Buffer[pBuf->RdIndx++];
453  }
454 
455  /* atomically adjust the NumAvailable */
456  ADI_DISABLE_INT(hDevice->UARTIRQn);
457  pBuf->NumAvailable -= Size;
458  ADI_ENABLE_INT(hDevice->UARTIRQn);
459  }
460 #endif /* (1 == ADI_UART_CFG_INTERRUPT_MODE_SUPPORT) */
461 
462 #if (1 == ADI_UART_CFG_POLLED_MODE_SUPPORT)
463  /* polling mode */
464  if(IS_POLLED_MODE(hDevice))
465  {
466  for(i=0; i < Size; i++)
467  {
468  /* blocking read */
469 
470  /* wait until a byte is available */
471  while(1)
472  {
473  /* read the line status register */
474  uint8_t status = hDevice->pUartRegs->COMLSR;
475 
476  /* Trap overflow, parity, frame errors */
477  if(status & (COMLSR_OE | COMLSR_PE | COMLSR_FE))
478  UART_RX_Status_Handler(hDevice, status);
479 
480  /* check if data is available */
481  if(status & COMLSR_DR)
482  break;
483  }
484 
485  /* read byte from the COMRX register */
486  *pRxData++ = hDevice->pUartRegs->COMRX;
487  /* increment the pointer */
488  }
489  }
490 #endif /* (1 == ADI_UART_CFG_POLLED_MODE_SUPPORT) */
491 
492  return(ADI_UART_SUCCESS);
493 }

508 static ADI_UART_RESULT_TYPE do_write(ADI_UART_HANDLE const hDevice, uint8_t *pTxData, const int16_t Size)
509 {
510 #if (1 == ADI_UART_CFG_INTERRUPT_MODE_SUPPORT)
511  if(IS_INTERRUPT_MODE(hDevice))
512  {
513  int16_t i = 0;
514  int16_t BytesToCopy = 0;
515  int16_t BytesRemaining= 0;
516  ADI_UART_CIRC_BUFFER_TYPE *pBuf = &hDevice->TxBuffer;
517 
518  i = pBuf->BufSize - pBuf->WrIndx;
519  BytesToCopy = (i > Size) ? Size : i;
520 
521  /* TODO: check if memcpy is faster */
522  for(i=0;i < BytesToCopy; i++)
523  pBuf->Buffer[pBuf->WrIndx++] = *pTxData++;
524 
525  /* check if we have more to copy */
526  if(BytesToCopy < Size)
527  {
528  BytesRemaining = Size - BytesToCopy;
529  pBuf->WrIndx = 0;
530 
531  for(i=0;i<BytesRemaining;i++)
532  pBuf->Buffer[pBuf->WrIndx++] = *pTxData++;
533  }
534 
535  /* atomically adjust the NumAvailable */
536  ADI_DISABLE_INT(hDevice->UARTIRQn);
537  pBuf->NumAvailable -= Size;
538 
539  /* and enable tx empty interrupt */
540  hDevice->pUartRegs->COMIEN |= COMIEN_ETBEI;
541  ADI_ENABLE_INT(hDevice->UARTIRQn);
542  }
543 #endif /* (1 == ADI_UART_CFG_INTERRUPT_MODE_SUPPORT) */
544 
545 #if (1 == ADI_UART_CFG_POLLED_MODE_SUPPORT)
546  if(IS_POLLED_MODE(hDevice))
547  {
548  for(int i=0; i < Size; i++)
549  {
550  /* blocking write */
551 
552  /* wait until previous byte is transmitted */
553  while(!(hDevice->pUartRegs->COMLSR & COMLSR_THRE));
554 
555  /* write byte to the COMTX register and increment the data pointer */
556  hDevice->pUartRegs->COMTX = *pTxData++;
557  }
558  }
559 #endif /* (1 == ADI_UART_CFG_POLLED_MODE_SUPPORT) */
560 
561  return(ADI_UART_SUCCESS);
562 }

579 static ADI_UART_RESULT_TYPE ProcessRequest(ADI_UART_HANDLE const hDevice,
580  uint8_t *pData,
581  int16_t *pSize,
582  bool_t bIsRxBuffer)
583 {
584  ADI_UART_RESULT_TYPE result = ADI_UART_ERR_UNKNOWN;
585 
586 #if (1 == ADI_UART_CFG_INTERRUPT_MODE_SUPPORT)
587  /* check if we are in interrupt mode */
588  if( IS_INTERRUPT_MODE(hDevice) )
589  {
590  ADI_UART_CIRC_BUFFER_TYPE *pCircBuffer = bIsRxBuffer ? &hDevice->RxBuffer : &hDevice->TxBuffer;
591 
592 #if (1 == ADI_UART_CFG_BLOCKING_MODE_SUPPORT)
593  /* blocking mode */
594  if( IS_BLOCKING_MODE(hDevice) )
595  {
596  if (pCircBuffer->BufSize < *pSize)
597  {
598  /* If the caller is requesting more data than the circular buffer
599  * can hold then the request can never be fulfilled.
600  */
601  result = ADI_UART_ERR_INVALID_BUFFER;
602  }
603  else
604  {
605  while(pCircBuffer->NumAvailable < *pSize)
606  {
607 #if (1 == ADI_CFG_ENABLE_RTOS_SUPPORT)
608  adi_osal_SemPend(hDevice->hSem, ADI_OSAL_TIMEOUT_FOREVER);
609 #else /* (0 == ADI_CFG_ENABLE_RTOS_SUPPORT) */
610  /* Instead of busy-waiting, we enter core sleep mode while waiting. Any
611  * interrupt that changes NumAvailable will also bring the processor
612  * out of core sleep and cause this call to return.
613  */
614  SystemEnterLowPowerMode(ADI_SYS_MODE_CORE_SLEEP, /* wait in core sleep */
615  &hDevice->bInterruptFlag,
616  0);
617 #endif /* (0 == ADI_CFG_ENABLE_RTOS_SUPPORT) */
618  }
619 
620  result = ADI_UART_SUCCESS;
621  }
622  }
623 #endif /* (1 == ADI_UART_CFG_BLOCKING_MODE_SUPPORT) */
624 
625 #if (1 == ADI_UART_CFG_NONBLOCKING_MODE_SUPPORT)
626  if( IS_NONBLOCKING_MODE(hDevice) ) /* non-blocking mode */
627  {
628  /* Adjust transfer size */
629  int16_t BytesAvailable = pCircBuffer->NumAvailable;
630 
631  *pSize = (*pSize > BytesAvailable) ? BytesAvailable : *pSize;
632  result = ADI_UART_SUCCESS;
633  }
634 #endif /* (1 == ADI_UART_CFG_NONBLOCKING_MODE_SUPPORT) */
635  }
636 #endif /* (1 == ADI_UART_CFG_INTERRUPT_MODE_SUPPORT) */
637 
638 #if (1 == ADI_UART_CFG_POLLED_MODE_SUPPORT)
639  if (IS_POLLED_MODE(hDevice)) /* polling mode */
640  {
641 #if (1 == ADI_UART_CFG_BLOCKING_MODE_SUPPORT)
642  /* in non-blocking mode if there is no data pSize is set to 0 */
643  if(IS_BLOCKING_MODE(hDevice))
644  {
645  /* No change to *pSize in polled-blocking mode, as we'll wait for the
646  * specified number of bytes to be received or transmitted.
647  */
648  result = ADI_UART_SUCCESS;
649  }
650 #endif /* (1 == ADI_UART_CFG_BLOCKING_MODE_SUPPORT) */
651 
652 #if (1 == ADI_UART_CFG_NONBLOCKING_MODE_SUPPORT)
653  if( IS_NONBLOCKING_MODE(hDevice) )
654  {
655  /* In polled-nonblocking mode, there can be at most one byte available */
656  *pSize = (bIsRxBuffer ? UART_RX_FULL(hDevice) : UART_TX_EMPTY(hDevice)) ? 1 : 0;
657  result = ADI_UART_SUCCESS;
658  }
659 #endif /* (1 == ADI_UART_CFG_NONBLOCKING_MODE_SUPPORT) */
660  }
661 #endif /* (1 == ADI_UART_CFG_POLLED_MODE_SUPPORT) */
662 
663  return(result);
664 }

739 ADI_UART_RESULT_TYPE adi_UART_Init(const ADI_UART_DEV_ID_TYPE devID, ADI_UART_HANDLE* const pHandle,
740  ADI_UART_INIT_DATA* const pInitData)
741 {
742  ADI_UART_HANDLE hDevice = NULL;
743  ADI_UART_RESULT_TYPE Result = ADI_UART_SUCCESS;
744 
745  /* set handle to null */
746  *pHandle = NULL;
747 
748 #if defined(ADI_DEBUG)
749  if(devID > NUM_UART_INSTANCES)
750  return (ADI_UART_ERR_INVALID_DEVID);
751 #endif
752 
753  /* address of the device data block is the handle */
754  hDevice = &UART_DevData[devID];
755 
756 #if defined(ADI_DEBUG)
757  /* if state is not unknown return error */
758  if( ADI_UART_DRV_STATE_UNKNOWN != hDevice->DrvState )
759  return(ADI_UART_ERR_ALREADY_INITIALIZED);
760 #endif /* defined(ADI_DEBUG) */
761 
762 #if (1 == ADI_UART_CFG_INTERRUPT_MODE_SUPPORT)
763  /* initialize uart using init data parameters */
764  if(pInitData != NULL)
765  {
766 #if defined(ADI_DEBUG)
767  /* check init parameters */
768  if( ( pInitData->pRxBufferData == NULL ) || (pInitData->RxBufferSize == 0) ||
769  ( pInitData->pTxBufferData == NULL ) || (pInitData->TxBufferSize == 0) )
770  return(ADI_UART_ERR_INVALID_BUFFER);
771 #endif /* defined(ADI_DEBUG) */
772 
773 #if (1 == ADI_CFG_ENABLE_RTOS_SUPPORT)
774  /* Create the semaphore for blocking mode support. */
775  if(adi_osal_SemCreateStatic(&hDevice->SemMemory, ADI_OSAL_MAX_SEM_SIZE_CHAR, &hDevice->hSem, 0u) != ADI_OSAL_SUCCESS)
776  {
777  return ADI_UART_ERR_SEMAPHORE_FAILED;
778  }
779 #else /* (0 == ADI_CFG_ENABLE_RTOS_SUPPORT) */
780  hDevice->bInterruptFlag = false;
781 #endif /* (0 == ADI_CFG_ENABLE_RTOS_SUPPORT) */
782 
783  /* initialize rx buffer parameters */
784  hDevice->RxBuffer.Buffer = pInitData->pRxBufferData;
785  hDevice->RxBuffer.BufSize = pInitData->RxBufferSize;
786 
787  /* initialize tx buffer parameters */
788  hDevice->TxBuffer.Buffer = pInitData->pTxBufferData;
789  hDevice->TxBuffer.BufSize = pInitData->TxBufferSize;
790 
791  /* enable interrupt mode if internal buffering is enabled */
792  hDevice->bInterruptMode = true;
793  ADI_INSTALL_HANDLER(hDevice->UARTIRQn, UART_Int_Handler);
794  }
795 #endif /* (1 == ADI_UART_CFG_INTERRUPT_MODE_SUPPORT) */
796 
797  ResetCircularBuffers(hDevice);
798 
799  /* set uart state to initialized */
800  hDevice->DrvState = ADI_UART_DRV_STATE_INITIALIZED;
801 
802  /* configure uart with default parameters */
803  adi_UART_SetBaudRate(hDevice, hDevice->defBaud);

1207 ADI_UART_RESULT_TYPE adi_UART_SetBaudRate(ADI_UART_HANDLE const hDevice, const ADI_UART_BAUDRATE_TYPE BaudRateIndex)
1208 {
1209  uint32_t incoming_clock;
1210 
1211  /* common baud rates and the associated comdiv, divm, divn values from the specification */
1212  /* other values can be optimized for minimal error based on HRM equation: */
1213  /* BaudRate = UCLK / (2*16*COMDIV) / (M+(N/2048)), */
1214  /* with: */
1215  /* 1<=COMDIV<=65536, */
1216  /* 1<=M<=3, and */
1217  /* 0<=N<=2047 */
1218  /* these constants are optimized for and valid only for either a 16MHz or 1MHz clock */
1219  typedef struct {
1220  uint16_t divC;
1221  uint16_t divM;
1222  uint16_t divN;
1223  } DIV_TYPE;
1224 
1225  /* create local divisor tables on stack */
1226  DIV_TYPE DivMap[][ADI_UART_BAUD_MAX_ENTRIES] = {
1227  {
1228  /* the 16MHz table... */
1229  { 49, 1, 129 }, /* 0.005937% error */
1230  { 19, 1, 759 }, /* 0.000625% error */
1231  { 4, 3, 523 }, /* 0.005000% error */
1232  { 8, 1, 174 }, /* 0.010001% error */
1233  { 3, 1, 915 }, /* 0.001250% error */
1234  { 2, 1, 174 }, /* 0.010001% error */
1235  { 1, 1, 174 }, /* 0.010001% error */
1236  },
1237  {
1238  /* the 1MHz table... */
1239  { 1, 3, 523 }, /* 0.005000% error */
1240  { 1, 1, 1285 }, /* 0.010001% error */
1241  { 1, 1, 0 }, /* 18.6% error, nonrealizable solution */
1242  { 1, 1, 0 }, /* 45.7% error, nonrealizable solution */
1243  { 1, 1, 0 }, /* 72.9% error, nonrealizable solution */
1244  { 1, 1, 0 }, /* 86.4% error, nonrealizable solution */
1245  { 1, 1, 0 }, /* 93.2% error, nonrealizable solution */
1246  },
1247 #ifndef ADI_ADUCM350 /* ADUCM350 only has maximum clock of 16 MHz */
1248  {
1249  /* the 20MHz table... */
1250  { 17, 3, 1699 }, /* 0.005937% error */
1251  { 10, 3, 523 }, /* 0.000625% error */
1252  { 7, 2, 666 }, /* 0.005000% error */
1253  { 3, 3, 1263 }, /* 0.010001% error */
1254  { 2, 2, 1460 }, /* 0.001250% error */
1255  { 1, 2, 1460 }, /* 0.010001% error */
1256  { 1, 1, 730 }, /* 0.010001% error */
1257  }
1258 #endif /* ADI_ADUCM350 */
1259  };
1260 
1261  /* pointer to table entries */
1262  DIV_TYPE *pMap;
1263 
1264 #if defined(ADI_DEBUG)
1265  /* check the instance */
1266  if( hDevice != &UART_DevData[ADI_UART_DEVID_0] )
1267  return(ADI_UART_ERR_INVALID_INSTANCE);
1268 
1269  if( hDevice->DrvState != ADI_UART_DRV_STATE_INITIALIZED )
1270  return ADI_UART_ERR_NOT_INITIALIZED;
1271 
1272  /* validate baudrate index */
1273  if (BaudRateIndex >= ADI_UART_BAUD_MAX_ENTRIES )
1274  return ADI_UART_ERR_INVALID_PARAMS;
1275 #endif /* defined(ADI_DEBUG) */
1276 
1277 
1278  /* get the system clock */
1279  // 'M350/360 have dedicated system clocks for UART
1280  incoming_clock = SystemGetClockFrequency(ADI_SYS_CLOCK_UART);
1281 
1282  /* only supporting 16MHz and 1MHz at this time... other clocks require new div table entries */
1283  if (incoming_clock == 16000000) { pMap = &DivMap[0][BaudRateIndex];
1284  } else if (incoming_clock == 1000000) { pMap = &DivMap[1][BaudRateIndex];
1285 #ifndef ADI_ADUCM350 /* ADUCM350 only has maximum clock of 16 MHz */
1286  } else if (incoming_clock == 20000000) { pMap = &DivMap[2][BaudRateIndex];
1287 #endif
1288  } else { return ADI_UART_ERR_INVALID_PARAMS; }
1289 
1290 
1291  hDevice->pUartRegs->COMDIV = pMap->divC; /* set the main baudrate divider */
1292  hDevice->pUartRegs->COMFBR = /* set the fractional baudrate dividers */
1293  pMap->divN | (pMap->divM << BITP_UART_COMFBR_DIVM) | COMFBR_ENABLE;
1294 
1295  return(ADI_UART_SUCCESS);
1296 }

1314 uint32_t adi_UART_GetBaudRate(const ADI_UART_HANDLE hDevice)
1315 {
1316  uint32_t incoming_clock;
1317  uint16_t ComDiv = hDevice->pUartRegs->COMDIV;
1318  uint16_t divN = hDevice->pUartRegs->COMFBR & COMFBR_DIVN_MSK;
1319  uint8_t divM = (hDevice->pUartRegs->COMFBR & COMFBR_DIVM_MSK) >> 11;
1320 
1321  /* get the system clock */
1322  // 'M350/360 have dedicated system clocks for UART
1323  incoming_clock = SystemGetClockFrequency(ADI_SYS_CLOCK_UART);
1324 
1325  /* Formula to compute Baud Rate = (UCLK) / ( 2 x 16 x (M + N/2048) * COMDIV )
1326  * rearranged to remove integer division, Baud Rate = (UCLK << 6) / (((M << 11) + N) * COMDIV)
1327  */
1328  return ((incoming_clock << 6) / ( ( (divM << 11) + divN) * ComDiv));
1329 }
