163 {
164 
165  ADI_GPIO_RESULT_TYPE Result;
166 
167  /* ever the optimist */
168  Result = ADI_GPIO_SUCCESS;
169 
170  /* only initialize on 1st init call, i.e., preserve callbacks on multiple inits */
171  if (!adi_GPIO_Device.ReferenceCount++) {
172 
173  /* (re)initialize the callback table */
174  for (int i = 0; i < ADI_GPIO_NUM_INTERRUPTS; i++)
175  {
176  adi_GPIO_Device.CallbackTable[i].pfCallback = NULL;
177  adi_GPIO_Device.CallbackTable[i].pCBParam = NULL;
178  }
179 
180  /* (re)install the handlers for external interrupts */
181  ADI_INSTALL_HANDLER(ADI_INT_EINT0, Ext_Int0_Handler);
182  ADI_INSTALL_HANDLER(ADI_INT_EINT1, Ext_Int1_Handler);
183  ADI_INSTALL_HANDLER(ADI_INT_EINT2, Ext_Int2_Handler);
184  ADI_INSTALL_HANDLER(ADI_INT_EINT3, Ext_Int3_Handler);
185  ADI_INSTALL_HANDLER(ADI_INT_EINT4, Ext_Int4_Handler);

187  ADI_INSTALL_HANDLER(ADI_INT_EINT6, Ext_Int6_Handler);
188  ADI_INSTALL_HANDLER(ADI_INT_EINT7, Ext_Int7_Handler);

268 ADI_GPIO_RESULT_TYPE adi_GPIO_ResetToPowerUp(void)
269 {
270 
271  ADI_GPIO_TypeDef *PortTable[] = { /* table of port base registers */
272  pADI_GP0, /* port 0 base address */
273  pADI_GP1, /* port 1 base address */
274  pADI_GP2, /* port 2 base address */
275  pADI_GP3, /* port 3 base address */
276  pADI_GP4, /* port 4 base address */
277  };
278 
279 
280  ADI_GPIO_TypeDef *pPort; /* pointer to port registers */
281  uint32_t i; /* counter */
282 
283 #ifdef ADI_DEBUG
284  /* make sure we're initialized */
285  if (adi_GPIO_Device.ReferenceCount == 0) {
286  return (ADI_GPIO_ERR_NOT_INITIALIZED);
287  }
288 #endif
289 
290  /* FOR (each port) */
291  for (i = 0; i < ((sizeof(PortTable))/(sizeof(PortTable[0]))); i++) {
292 
293  /* point to the register block */
294  pPort = PortTable[i];
295 
296  /* reset the registers */
297  pPort->GPCON = ADI_GPIO_DEFAULT_GPCON;
298  pPort->GPOEN = ADI_GPIO_DEFAULT_GPOEN;
299 
300  /* 350 has single pull enable register for either hard-wired pull direction */
301  if (pPort == pADI_GP0)
302  {
303  pPort->GPPE = ADI_GPIO_GP0PE_RESET; /* For Port0 don't configure JTAG/SWD pins */
304  }
305  else
306  {
307  pPort->GPPE = ADI_GPIO_DEFAULT_GPPE;
308  }
309 
310  /* additional interrupt registers for M350 */
311  pPort->GPIPOL = ADI_GPIO_DEFAULT_GPIPOL;
312  pPort->GPIENA = ADI_GPIO_DEFAULT_GPIENA;
313  pPort->GPIENB = ADI_GPIO_DEFAULT_GPIENB;
314  pPort->GPISTAT = ADI_GPIO_DEFAULT_GPISTAT;
315 
316  /* ENDFOR */
317  }
318 
319  /* return */
320  return (ADI_GPIO_SUCCESS);
321 }

390 ADI_GPIO_RESULT_TYPE adi_GPIO_EnableIRQ (const IRQn_Type eIrq, const ADI_GPIO_IRQ_TRIGGER_CONDITION_TYPE eCondition)
391 {
392 
393  uint16_t volatile *pIRQCFG; /* pointer to the appropriate interrupt config register */
394  uint16_t Mask; /* mask to manipulate the register */
395  uint16_t Pattern; /* bit pattern that will be written into the register */
396  uint32_t ShiftCount; /* shift count */
397  uint16_t CfgReg; /* interrupt config register value */
398 
399 #ifdef ADI_DEBUG
400  /* make sure we're initialized */
401  if (adi_GPIO_Device.ReferenceCount == 0) {
402  return (ADI_GPIO_ERR_NOT_INITIALIZED);
403  }
404 
405  /* validate the interrupt number */
406  if (!IsInterruptValid(eIrq)) {
407  return (ADI_GPIO_ERR_INVALID_INTERRUPT);
408  }
409 
410  /* validate the trigger condition */
411  if (!IsTriggerValid(eCondition)) {
412  return (ADI_GPIO_ERR_INVALID_TRIGGER);
413  }
414 #endif
415 
416  /* figure out the shift count and which config register we need to change */
417  switch (eIrq) {
418  case EINT0_IRQn:
419  case EINT1_IRQn:
420  case EINT2_IRQn:
421  case EINT3_IRQn:
422  /* external interrupts 0-3 */
423  pIRQCFG = &pADI_IDU->EI0CFG;
424  ShiftCount = (eIrq - EINT0_IRQn) << ADI_GPIO_EIxCFG_SHIFT_COUNT;
425  break;
426 
427  case EINT4_IRQn:
428  case EINT5_IRQn:
429  case EINT6_IRQn:
430  case EINT7_IRQn:
431  /* external interrupts 4-7 */
432  pIRQCFG = &pADI_IDU->EI1CFG;
433  ShiftCount = (eIrq - EINT4_IRQn) << ADI_GPIO_EIxCFG_SHIFT_COUNT;
434  break;
435 
436  case EINT8_IRQn:
437  /* external interrupt 8 */
438  pIRQCFG = &pADI_IDU->EI2CFG;
439  ShiftCount = (eIrq - EINT8_IRQn) << ADI_GPIO_EIxCFG_SHIFT_COUNT;
440  break;
441 
442  default:
443  return ADI_GPIO_ERR_INVALID_INTERRUPT;
444  }
445 
446  /* create the mask we'll use to clear the relevant bits in the config register */
447  Mask = (EI0CFG_IRQ0MDE_MSK | EI0CFG_IRQ0EN) << ShiftCount;
448 
449  /* create the bit pattern we're going to write into the configuration register */
450  Pattern = (EI0CFG_IRQ0EN | eCondition) << ShiftCount;
451 
452  /*********** need to protect this sequence *****/
453  ADI_ENTER_CRITICAL_REGION();
454 
455  /* read/modify/write the appropriate bits in the register */
456  CfgReg = *pIRQCFG;
457  CfgReg &= ~Mask;
458  CfgReg |= Pattern;
459  *pIRQCFG = CfgReg;
460 
461  /*********** turn off protection *****/
462  ADI_EXIT_CRITICAL_REGION();
463 
464  /* enable the interrupt */
465  NVIC_EnableIRQ(eIrq);
466 
467  /* return */
468  return (ADI_GPIO_SUCCESS);
469 }

492 ADI_GPIO_RESULT_TYPE adi_GPIO_DisableIRQ (const IRQn_Type eIrq)
493 {
494 
495  uint16_t volatile *pIRQCFG; /* pointer to the appropriate interrupt config register */
496  uint16_t Mask; /* mask to manipulate the register */
497  uint32_t ShiftCount; /* shift count */
498  uint16_t CfgReg; /* interrupt config register value */
499 
500 #ifdef ADI_DEBUG
501  /* make sure we're initialized */
502  if (adi_GPIO_Device.ReferenceCount == 0) {
503  return (ADI_GPIO_ERR_NOT_INITIALIZED);
504  }
505 
506  /* validate the interrupt number */
507  if (!IsInterruptValid(eIrq)) {
508  return (ADI_GPIO_ERR_INVALID_INTERRUPT);
509  }
510 #endif
511 
512  /* disable the interrupt */
513  NVIC_DisableIRQ(eIrq);
514 
515  /* figure out the shift count and which config register we need to change */
516  switch (eIrq) {
517  case EINT0_IRQn:
518  case EINT1_IRQn:
519  case EINT2_IRQn:
520  case EINT3_IRQn:
521  /* external interrupts 0-3 */
522  pIRQCFG = &pADI_IDU->EI0CFG;
523  ShiftCount = (eIrq - EINT0_IRQn) << ADI_GPIO_EIxCFG_SHIFT_COUNT;
524  break;
525 
526  case EINT4_IRQn:
527  case EINT5_IRQn:
528  case EINT6_IRQn:
529  case EINT7_IRQn:
530  /* external interrupts 4-7 */
531  pIRQCFG = &pADI_IDU->EI1CFG;
532  ShiftCount = (eIrq - EINT4_IRQn) << ADI_GPIO_EIxCFG_SHIFT_COUNT;
533  break;
534 
535  case EINT8_IRQn:
536  /* external interrupt 8 */
537  pIRQCFG = &pADI_IDU->EI2CFG;
538  ShiftCount = (eIrq - EINT8_IRQn) << ADI_GPIO_EIxCFG_SHIFT_COUNT;
539  break;
540 
541  default:
542  return ADI_GPIO_ERR_INVALID_INTERRUPT;
543  }
544 
545  /* create the mask we'll use to clear the interrupt enable bit in the config register */
546  Mask = EI0CFG_IRQ0EN << ShiftCount;
547 
548  /*********** need to protect this sequence *****/
549  ADI_ENTER_CRITICAL_REGION();
550 
551  /* read/modify/write the appropriate bits in the register */
552  CfgReg = *pIRQCFG;
553  CfgReg &= ~Mask;
554  *pIRQCFG = CfgReg;
555 
556  /*********** turn off protection *****/
557  ADI_EXIT_CRITICAL_REGION();
558 
559  /* return */
560  return (ADI_GPIO_SUCCESS);
561 }

1774 /* Common external interrupt handler */
1775 void CommonInterruptHandler(const ADI_GPIO_IRQ_INDEX_TYPE index, const IRQn_Type eIrq)
1776 {
1777  GPIO_CALLBACK_INFO *pCallbackInfo = &adi_GPIO_Device.CallbackTable[index];
1778  /* GroupA/B interrupt clearing variables unique to ADiCM350 */
1779  ADI_GPIO_TypeDef *pPorts[] = { pADI_GP0, pADI_GP1, pADI_GP2, pADI_GP3, pADI_GP4 };
1780  ADI_GPIO_TypeDef *pPort;
1781  ADI_GPIO_DATA_TYPE interruptingPins, enabledPins;
1782 
1783  /* First dispatch the callback, if one is registered */
1784  if(pCallbackInfo->pfCallback) {
1785  /* params list is: application-provided cbParam, interrupt ID, and NULL */
1786  pCallbackInfo->pfCallback (pCallbackInfo->pCBParam, (uint32_t) eIrq, NULL);
1787  }
1788 
1789  /* now clear the interrupt... */
1790  switch (eIrq) {
1791  /* GroupA/B interrupts take special handeling to clear (only on ADuCM350) */
1792  case GPIOA_IRQn:
1793  case GPIOB_IRQn:
1794 
1795  /* collect interrupting pins over each port and clear them */
1796  for (int i = 0; i < sizeof(pPorts)/sizeof(pPorts[0]); i++) {
1797  pPort = pPorts[i];
1798 
1799  /* which interruptingPins are enabled (by group) */
1800  enabledPins = (GPIOA_IRQn == eIrq) ? pPort->GPIENA : pPort->GPIENB;
1801 
1802  /* clear enabled pins that actually interrupted, if any (for this port and group) */
1803  interruptingPins = ((pPort->GPINT) & enabledPins);
1804  if (interruptingPins)
1805  pPort->GPINT = interruptingPins;
1806  }
1807  break;
1808  /* otherwise, just a simple clear */
1809  default:
1810  adi_GPIO_ClearIRQ(eIrq);
1811  break;
1812  }
1813 }
