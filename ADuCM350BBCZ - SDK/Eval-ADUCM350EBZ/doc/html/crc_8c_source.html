622 ADI_CRC_RESULT_TYPE adi_CRC_SetDmaMode(ADI_CRC_DEV_HANDLE const hDevice,const bool_t bFlag)
623 {
624 
625  ADI_CRC_RESULT_TYPE result = ADI_CRC_SUCCESS;
626 
627 #ifdef ADI_DEBUG
628  if (ADI_CRC_INVALID_HANDLE(hDevice))
629  return ADI_CRC_ERR_BAD_DEV_HANDLE;
630 
631  if (hDevice->InitState != ADI_CRC_STATE_INITIALIZED)
632  return ADI_CRC_ERR_NOT_INITIALIZED ;

634  if(bFlag == true)
635  {
636  /* Enable the CRC DMA interrupt */
637  ADI_ENABLE_INT(DMA_CRC_IRQn);
638  hDevice->pDevice->CRC_CTL |= (BITM_CRC_CRC_CTL_DMAREQEN);
639  /* Initialize DMA service */
640  if (ADI_DMA_SUCCESS != adi_DMA_Init(CRC_CHANn, ADI_DMA_PRIORITY_RESET))
641  return ADI_CRC_ERR_BAD_DMA_CONFIG;
642  /* Install the DMA interrupt handler */
643  ADI_INSTALL_HANDLER(DMA_CRC_IRQn, DMA_CRC_Int_Handler);
644 
645  }
646  else
647  {
648  /* Disable CRC DMA IRQ */
649  ADI_DISABLE_INT(DMA_CRC_IRQn);
650  hDevice->pDevice->CRC_CTL &= ~(BITM_CRC_CRC_CTL_DMAREQEN);
651  /* Uninit the DMA service for CRC DMA channel */
652  if (ADI_DMA_SUCCESS != adi_DMA_UnInit(CRC_CHANn))
653  return ADI_CRC_ERR_BAD_DMA_CONFIG;
654  /* Uninstall the DMA interrupt handler for CRC DMA channel */
655  ADI_UNINSTALL_HANDLER(DMA_CRC_IRQn);
656  }
657  return result;
658 }

678 ADI_CRC_RESULT_TYPE adi_CRC_BufferSubmit(ADI_CRC_DEV_HANDLE const hDevice,uint32_t *pBuff, uint32_t nNumWords)
679 {
680 #ifdef ADI_DEBUG
681  if (ADI_CRC_INVALID_HANDLE(hDevice))
682  return ADI_CRC_ERR_BAD_DEV_HANDLE;
683 
684  if (hDevice->InitState != ADI_CRC_STATE_INITIALIZED)
685  return ADI_CRC_ERR_NOT_INITIALIZED ;
686 
687  if (((uint32_t)pBuff&0X03) != 0)
688  return ADI_CRC_ERR_BAD_DATA_POINTER ;
689 
690 #endif
691  /* Device is calculating the CRC for previously submitted buffer */
692  if(hDevice->bCRCBusy == true)
693  {
694  return(ADI_CRC_ERR_DEVICE_BUSY);
695  }
696  /* store the size of the buffer */
697  hDevice->unscheduledCnt = nNumWords;
698  /* Can the buffer processed in one iteration */
699  if(nNumWords >= ADI_DMA_MAX_TRANSFER_SIZE)
700  {
701  nNumWords = ADI_DMA_MAX_TRANSFER_SIZE;
702  /* Buffer address for next iteration */
703  hDevice->pBuffer = pBuff+ ADI_DMA_MAX_TRANSFER_SIZE;
704  hDevice->unscheduledCnt-= ADI_DMA_MAX_TRANSFER_SIZE;
705  }
706  else
707  {
708  /* As DMA for the complete buffer will be scheduled remaining elements
709  for next iteration will be 0 */
710  hDevice->unscheduledCnt = 0x0;
711 
712  /* Only one iteration */
713  hDevice->pBuffer = pBuff;
714  }
715  /*Initialize the address in DMA descriptor*/
716  gDmaDescriptor.pSrcData = pBuff;
717  /*Store the number of transfer */
718  gDmaDescriptor.DataLength = nNumWords;
719  /*Mark the status of the device as BUSY */
720  hDevice->bCRCBusy = true;
721  /* submit the buffer to DMA service */
722  if (adi_DMA_SubmitTransfer(&gDmaDescriptor) != ADI_DMA_SUCCESS) {
723 #ifdef ADI_DEBUG
724  /* Reset if the submission is failed */
725  hDevice->pBuffer = NULL;
726  hDevice->unscheduledCnt = 0X00;
727 
728 #endif
729  hDevice->bCRCBusy = false;
730  return ADI_CRC_ERR_BAD_DMA_CONFIG;
731  }
732  return ADI_CRC_SUCCESS;}

928 ADI_INT_HANDLER(DMA_CRC_Int_Handler)
929 {
930  uint32_t nSize;
931  /* Check for the remaining words */
932  if(pCrc_DevData->unscheduledCnt > 0)
933  {
934  /* Calculate the transfer size for next iteration */
935  nSize =MIN(pCrc_DevData->unscheduledCnt, ADI_DMA_MAX_TRANSFER_SIZE);
936  /* update the descriptor */
937  gDmaDescriptor.pSrcData = pCrc_DevData->pBuffer;
938  gDmaDescriptor.DataLength = nSize;
939  gDmaDescriptor.Mode = ADI_DMA_MODE_BASIC;
940  /* Submit to DMA manager*/
941  adi_DMA_SubmitTransfer(&gDmaDescriptor);
942  /* Update the pointer for next iteration.*/
943  pCrc_DevData->pBuffer += nSize;
944  pCrc_DevData->unscheduledCnt -= nSize;
945  }
946  else /* No more iteration. Stop the DMA. Reset the pointers */
947  {
948  ADI_ENTER_CRITICAL_REGION();
949  gDmaDescriptor.pSrcData = NULL;
950  gDmaDescriptor.DataLength = 0;
951  pCrc_DevData->bCRCBusy = false;
952  pCrc_DevData->pBuffer = NULL;
953  ADI_EXIT_CRITICAL_REGION();
954 #if (ADI_CFG_ENABLE_CALLBACK_SUPPORT == 1)
955  /* call the registered callback function */
956  if(pCrc_DevData->pfCallback != NULL)
957  {
958  pCrc_DevData->pfCallback(pCrc_DevData->pCBParam, (uint32_t) ADI_CRC_EVENT_RESULT_READY, (void *)pCrc_DevData->pCRC->CRC_RESULT);
959  }
