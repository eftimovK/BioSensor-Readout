508 ADI_I2C_RESULT_TYPE adi_I2C_UnInit(ADI_I2C_DEV_HANDLE const hDevice) {
509 
510 #ifdef ADI_DEBUG
511  if (ADI_I2C_INVALID_HANDLE(hDevice))
512  return ADI_I2C_ERR_BAD_DEVICE_HANDLE;
513 
514  if (ADI_I2C_STATE_UNINITIALIZED == hDevice->initState)
515  return ADI_I2C_ERR_NOT_INITIALIZED;
516 
517  /* protect pending transaction that may not be fully serialized yet */
518  if ( (hDevice->initState == ADI_I2C_STATE_MASTER_MODE && MASTER_BUSY)
519 #if (ADI_I2C_CFG_SLAVE_MODE_SUPPORT == 1)
520  || (hDevice->initState == ADI_I2C_STATE_SLAVE_MODE && SLAVE_BUSY)
521 #endif /* ADI_I2C_CFG_SLAVE_MODE_SUPPORT */
522  )
523  return ADI_I2C_ERR_BUSY;
524 #endif
525 
526  /* disable bus error capture */
527  ADI_DISABLE_INT(DMA_ERR_IRQn);
528 
529  /* disable NVIC */
530  configureNVIC(hDevice, false);
531 
532  /* reset I2C control registers */
533  hDevice->pDev->I2CMCON = I2CMSTA_RVAL;
534  hDevice->pDev->I2CSCON = I2CSSTA_RVAL;
535 
536 #if (ADI_CFG_ENABLE_RTOS_SUPPORT == 1)
537  /* destroy semaphore */
538  if (adi_osal_SemDestroyStatic(hDevice->hSem) != ADI_OSAL_SUCCESS)
539  {
540  return ADI_I2C_ERR_SEMAPHORE_FAILED;
541  }
542 #endif /* ADI_CFG_ENABLE_RTOS_SUPPORT */
543 
544  /* call uninitialize helper */
545  if (ADI_I2C_SUCCESS != hDevice->FunctionTable.pfUnInitFcn(hDevice))
546  return ADI_I2C_ERR_BAD_DMA_CONFIG;
547 
548  hDevice->initState = ADI_I2C_STATE_UNINITIALIZED;
549 
550  return ADI_I2C_SUCCESS;
551 }

1860 static void CommenceMasterTransfer(ADI_I2C_DEV_HANDLE const hDevice) {
1861  /*
1862  Initiates a master-mode transfer.
1863  Program the master control byte with control code, chip select bits (slaveID)
1864  and r/w bit. Call this AFTER device enable because writing ADR0 is what actually
1865  initiates the transfer. Note that in 10-bit addressing mode, ADR1 write MUST occur
1866  BEFORE writing ADR0.
1867 
1868  Note that repeat starts are generated by rewriting the first master address byte register in
1869  the first master-mode TX/Rx interrupt... so save it as firstMasterAddressByte for later.
1870  */
1871 
1872  if (hDevice->bTenBitSlaveAddress) {
1873 
1874  /* 10-bit slave addressing spans two registers... */
1875 
1876  /* 2nd addr reg gets the lower-order 8 bits of slave ID address */
1877  SECOND_MAB = hDevice->slaveID & 0x00FF;
1878 
1879  /* 1st addr reg gets "0000000011110xxy"b + upper two ID bits (9:8) upshifted by one (xx) + r/w bit in LS position (y) */
1880  hDevice->firstMasterAddressByte = ( 0x00F0 | ((hDevice->slaveID & 0x0300) >> (8-1)) )
1881  | (hDevice->bMasterReadBit ? 0x0001 : 0x00000);
1882 
1883  } else {
1884 
1885  /* seven-bit address (upshifted by one) + w/r bit */
1886  hDevice->firstMasterAddressByte = ((hDevice->slaveID & 0x007F) << 1) | (hDevice->bMasterReadBit ? 0x0001 : 0x00000);
1887  }
1888 
1889  hDevice->bComplete = false;
1890 
1891 #if (ADI_I2C_CFG_ENABLE_DMA_SUPPORT == 1)
1892  hDevice->bDmaComplete = false;
1893 #endif /* ADI_I2C_CFG_ENABLE_DMA_SUPPORT */
1894 
1895  /* this is the actual launch */
1896  FIRST_MAB = hDevice->firstMasterAddressByte;
1897 }
