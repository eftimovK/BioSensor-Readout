20 /*
21 ATTRIBUTE_INTERRUPT
22  If this is available for a compiler, flag that a function is an interrupt
23  function. This can allow a compiler do some special handling.
24 KEEP_VAR(var)
25  If this is available for a compiler, flag that "var" should NOT be
26  optimised away by the compiler even if the compiler thinks that it is not
27  used.
28 WEAK_PROTO(proto)
29  If this is available for a compiler, apply whatever attributes are needed
30  to a function prototype ("proto") to flag that the function is a "weak" one.
31 WEAK_FUNC(func)
32  If this is available for a compiler, apply whatever attributes are needed
33  to a function definition ("func") to flag that the function is a "weak" one.
34 VECTOR_SECTION
35  A particular setup may have a requirement that the vector table be placed
36  in a particular section. This specifies the name of that section
37 SECTION_PLACE(def,sectionname)
38  Place the "def" variable in the named section.
39 RESET_EXCPT_HNDLR
40  A particular setup may have a requirement for a different reset handler.
41  This specifies the name of that handler.
42 ALIGN_VAR
43  Align a variable to a specific alignment. Note that some compilers do this
44  using pragma which can't be included in a macro expansion.(IAR)
45 */

108 //*****************************************************************************
109 //
110 // Local helpers for consistent ISR prototypes and implementations.
111 //
112 // The only Interrupts that should vector through the OSAL are the driver interrupts for which
113 // ADI has provided drivers which can operate in both an RTOS and a non-RTOS context.
114 //
115 // When operating in an RTOS context interrupts are vectord from the NVIC controller
116 // to a second level dispatch function known as the OSAL (operating system abstraction layer) dispatch.
117 // The OSAL dispatch has common code that will interact with the RTOS.
118 //
119 // OSAL dispatched interrupts must pass their Interrupt ID (iid) to the common OSAL dispatch.
120 //
121 // System interrupts (Reset through SysTick) should never dispatch through the OSAL.
122 //
123 // The WEAK_PROTOTYPE and WEAK_FUNCTION macros are used for System interrupts handlers. These
124 // handlers will always have a 'void' paramater list.
125 //
126 // The ADI_WEAK_PROTOTYPE and ADI_WEAK_FUNCTION macros will change the declaration of the Driver Interrupts Handlers
127 // depending on whether or not they are operating in an RTOS context. In an RTOS context the IID (interrupy ID)
128 // paramater and a driver specific 'void *' paramater are required. In a non-RTOS context there is no OSAL dispatch
129 // and the paramater list is simply 'void'.
130 //
131 //*****************************************************************************
132 #define WEAK_PROTOTYPE(x) WEAK_PROTO ( void x (void)) ATTRIBUTE_INTERRUPT ;
133 #define WEAK_FUNCTION(x) WEAK_FUNC ( void x (void)) { while(1); }
134 
135 #if (ADI_CFG_ENABLE_RTOS_SUPPORT == 1)
136 #define ADI_WEAK_PROTOTYPE(x) WEAK_PROTO ( void x (uint32_t, void*)) ATTRIBUTE_INTERRUPT ;
137 #define ADI_WEAK_FUNCTION(x) WEAK_FUNC ( void x (uint32_t iid, void* handlerArg)) { while(1); }
138 #define ADI_WEAK_FUNCTION2(x) WEAK_FUNC ( void x (uint32_t iid, void* handlerArg)) { }
139 #else
140 #define ADI_WEAK_PROTOTYPE(x) WEAK_PROTO ( void x (void)) ATTRIBUTE_INTERRUPT ;
141 #define ADI_WEAK_FUNCTION(x) WEAK_FUNC ( void x (void)) { while(1); }
142 #endif
