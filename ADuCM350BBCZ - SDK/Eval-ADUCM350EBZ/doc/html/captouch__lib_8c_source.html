55 ADI_CT_RESULT_TYPE adi_CT_Stage_AutoCAPDAC_Adjustment(ADI_CT_DEV_HANDLE const hDevice,ADI_CT_STAGE_NUMBER_TYPE StageNumber,uint16_t BaselineValue){
56 
57 
58  /* Stage Configuration Address Pointer */
59  volatile uint32_t *StageConfigurationPtr = NULL;
60 
61 
62  /* VARIABLES TO BE USED IN THIS CODE*/
63 
64 
65  /* To save the Stage Configuration */
66  uint32_t StageConfiguration;
67 
68  //new test
69  uint16_t bestdifference = 0xFFFF;
70  uint16_t newdifference = 0;
71  uint32_t StageConfigurationBestCfg;
72 
73  /* Lower and Upper Limits for the Baseline Desired Value */
74  uint16_t LowBaselineValue = BaselineValue - DEFAULT_BASELINE_LIMITS;
75  uint16_t UpBaselineValue = BaselineValue + DEFAULT_BASELINE_LIMITS;
76 
77  /* CDC Stage Value */
78  uint16_t CDCStageValue;
79 
80  /* Notify that the Baseline Value is set */
81  bool_t StageFinish=0;
82 
83  /* Initialize the Stage Addres Pointer */
84  StageConfigurationPtr = &pADI_CT->CT_STAGE0_CFG + StageNumber;
85 
86  /***************************************************************
87  It is necessary change some specific parameters to make the adjust
88  correctly. So, the system stores these parameters values to
89  reconfigure the Captouch once the CAPDAC adjust will be done
90  ***************************************************************/
91 
92  /* To store the Previous Captouch Configuration */
93  uint8_t PreviousCaptouchInterrupts = pADI_CT->CT_IEN;
94 
95  /* To store the Previous IIR Filter Status */
96  bool_t PreviousFilterStatus;
97 
98  /* To store the Previous IIR Filter Status */
99  bool_t PreviousSelfTimerStatus;
100 
101  /* To store the Previous Touch And Release Detection Status */
102  bool_t PreviousTouchAndReleaseDetectionStatus;
103 
104  /* To store the Previous Power Status */
105  bool_t PreviousPowerStatus;
106 
107  /* Store Enabled Interrupts */
108  adi_CT_GetInterruptsEnable(hDevice,&PreviousCaptouchInterrupts);
109 
110  /* Store the IIR Filter Status */
111  adi_CT_GetIIRFilterEnable(hDevice,&PreviousFilterStatus);
112 
113  /* Store the Self Timer Status */
114  adi_CT_GetSelfTimerEnable(hDevice,&PreviousSelfTimerStatus);
115 
116  /* Store the Touch And Release Detection Status */
117  adi_CT_GetTouchEnable(hDevice,&PreviousTouchAndReleaseDetectionStatus);
118 
119  /* Store the Touch And Release Detection Status */
120  adi_CT_GetPowerStatus(hDevice,&PreviousPowerStatus);
121 
122  /* Disable the IIR FIlter */
123  adi_CT_SetIIRFilterEnable(hDevice,false);
124 
125  /* Disable the Self Timer */
126  adi_CT_SetSelfTimerEnable(hDevice,false);
127 
128  /* Disable the Touch And Release Detection */
129  adi_CT_SetTouchEnable(hDevice,false);
130  adi_CT_SetReleaseEnable(hDevice,false);
131 
132  /* Disable the Captouch Interrupts */
133  adi_CT_SetInterruptsEnable(hDevice,PreviousCaptouchInterrupts,false);
134 
135  /* Reset the Main DAC and SUB DAC value */
136  *StageConfigurationPtr &= ~(BITM_CT_CT_STAGE1_CFG_MAIN_DAC | BITM_CT_CT_STAGE1_CFG_SUB_DAC);
137 
138  // *StageConfigurationPtr |= 0x04;
139 
140  /* Enable the Conversion Complete Interrupt */
141  adi_CT_SetInterruptsEnable(hDevice,ADI_CT_INTERRUPT_CONV_SET_COMPLETE,true);
142 
143 
144  /* Power Up the Captouch */
145  adi_CT_SetPowerStatus(hDevice,true);
146 
147 
148  /* For loop with 10 iterations (the same than the number of CAPDAC bits) */
149  uint8_t i = 10;
150 
151  /* Conditions to finish the loop. 10 iterations or Finish the CAPDAC Adjust */
152  while((i >= 1) && (StageFinish == 0))
153 
154  {
155  /* Stores the current Stage Configuration */
156  StageConfiguration = *StageConfigurationPtr;
157 
158  /* Change the CAPDAC value setting the next bit to 1 */
159  *StageConfigurationPtr |= (1 << i-1);
160 
161  /* Start a single conversion sequence and wait until it finishes (detected by CONV_SET_COMPLETE Interrupt) */
162  adi_CT_SetStartSingleSequence(hDevice);
163 
164  while (pADI_CT->CT_CDC_PWR & BITM_CT_CT_CDC_PWR_SW_STATUS);
165 
166  /* Read the CDC Value */
167  adi_CT_GetCDCValue(hDevice,(ADI_CT_STAGE_NUMBER_TYPE)StageNumber,&CDCStageValue);
168 
169  /* Check if the current CDC Value in between the limits */
170  if(CDCStageValue <= UpBaselineValue && CDCStageValue >= LowBaselineValue)
171  {
172  /* Adjust Finished */
173  StageFinish = 1;
174  StageConfigurationBestCfg = *StageConfigurationPtr; // Store the current configuration as the best
175  }
176 
177  else if(CDCStageValue > UpBaselineValue)
178  {
179 
180  /* Calculate the new difference */
181  newdifference = CDCStageValue - UpBaselineValue;
182  if(newdifference < bestdifference)
183  {
184  StageConfigurationBestCfg = *StageConfigurationPtr;
185  bestdifference = newdifference;
186  }
187  /* As the value is greater than the limit, the system doesn't store the last bit added */
188  *StageConfigurationPtr = StageConfiguration;
189 
190  }
191 
192  else //(CDCStageValue < LowBaselineValue)
193  {
194  newdifference = LowBaselineValue-CDCStageValue;
195  if(newdifference < bestdifference)
196  {
197  StageConfigurationBestCfg = *StageConfigurationPtr;
198  bestdifference = newdifference;
199  }
200  }
201 
202  /* Decrement the counter variable */
203  i--;
204 
205 
206  }
207 
208  *StageConfigurationPtr = StageConfigurationBestCfg;
209 
210  /* Once the CAPDAC is adjusted the system has to restablish the previous values */
211 
212  /* Set the Touch And Release Previous Status */
213  adi_CT_SetTouchEnable(hDevice,PreviousTouchAndReleaseDetectionStatus);
214  adi_CT_SetReleaseEnable(hDevice,PreviousTouchAndReleaseDetectionStatus);
215 
216  /* Set the Touch And Release Previous Status */
217  adi_CT_SetIIRFilterEnable(hDevice,PreviousFilterStatus);
218 
219  /* Set the Previous Self Timer Status */
220  adi_CT_SetSelfTimerEnable(hDevice,PreviousSelfTimerStatus);
221 
222  /* Disable the CON_SET_COMPLETE interrupt */
223  adi_CT_SetInterruptsEnable(hDevice,ADI_CT_INTERRUPT_CONV_SET_COMPLETE,false);
224 
225  /* Enable the Previous Captouch Interrupt */
226  adi_CT_SetInterruptsEnable(hDevice,PreviousCaptouchInterrupts,true);
227 
228  /* Set the Power Previous Status */
229  adi_CT_SetPowerStatus(hDevice,PreviousPowerStatus);
230 
231  /* First of all, the systems forces an autocalibration */
232  adi_CT_SetAutoForceBaselineCal(hDevice);
233 
234  return ADI_CT_RESULT_SUCCESS;
235 
236 }

255 ADI_CT_RESULT_TYPE adi_CT_FrequencyHoppingAlgorithm(ADI_CT_DEV_HANDLE const hDevice,ADI_CT_STAGE_NUMBER_TYPE StageNumber,uint16_t NoiseLevel,uint8_t *Phase24Result)
256 {
257 
258  #define M 10
259 
260  ADI_CT_C2V_TIMING_TYPE TimingSettings;
261 
262  #define MIN_PHASE24 0x10 // 1us
263  #define MAX_PHASE24 0x40 // 8us
264 
265  uint32_t MeasuredNoise;
266 
267  uint32_t NoiseFloor = 0;
268  uint32_t MaxNoiseValue = 0;
269  uint32_t MinNoiseValue = 0xFFFFFFFF;
270 
271  bool_t minimum = false;
272 
273  uint8_t CurrentPhase = MIN_PHASE24;
274  uint8_t MinNoisePhase;
275 
276 
277  /* get the current timing settings*/
278  adi_CT_GetTimingsCfg(hDevice,&TimingSettings);
279  TimingSettings.Phase24 = CurrentPhase;
280 
281 
282 
283  /***************************************************************
284  It is necessary change some specific parameters to make the adjust
285  correctly. So, the system stores these parameters values to
286  reconfigure the Captouch once the CAPDAC adjust will be done
287  ***************************************************************/
288 
289 
290  minimum = false;
291  while(minimum == false && CurrentPhase < MAX_PHASE24)
292 
293  {
294 
295  /* Measure the Noise*/
296  adi_CT_NoiseMeasurementPK2PK(hDevice,StageNumber,&MeasuredNoise,5);
297 
298 
299  /* Compare the current noise value with the "noise floor"*/
300  if(MeasuredNoise > NoiseFloor)
301  {
302  /* define a new noise floor */
303  NoiseFloor = MeasuredNoise;
304  }
305 
306  if (MeasuredNoise > MaxNoiseValue)
307  {
308  /* A noiser phase has been found, keep record of it*/
309  MaxNoiseValue = MeasuredNoise;
310  }
311 
312  /* Store the first phase time as the best one in order not to miss a
313  notch at the initial phase time */
314  if(CurrentPhase == MIN_PHASE24)
315  {
316  MinNoiseValue = MeasuredNoise;
317  MinNoisePhase = CurrentPhase;
318  }
319  else
320  {
321  /* compare the current noise with the minimum*/
322  if(MeasuredNoise < MinNoiseValue)
323  {
324  /* A less noisy phase time has been found, meaning that we are going into a local minimum */
325  MinNoiseValue = MeasuredNoise;
326  MinNoisePhase = CurrentPhase;
327  }
329  else
330  {
331  /* See if the possible local minimum is a proper local minimum */
332  /* if( ((NoiseFloor - MinNoiseValue) >= LocalMinimumThreshold) && ((MeasuredNoise - MinNoiseValue) >= LocalMinimumThreshold))
333  {
334  CurrentPhase = MinNoisePhase;
335  minimum = true;
336  }*/
337 
338  }
340  }
341 
342  if(MinNoiseValue <= NoiseLevel)
343  {
344  minimum = true;
345  }
346 
347  if( minimum == false)
348  {
349  /* Increment the phase time */
350  if(CurrentPhase++ >= MAX_PHASE24)
351  {
352  /* if the phase is bigger than the MAX_PHASE24 the algorithm stops*/
353  TimingSettings.Phase24 = MinNoisePhase;
354  }
355  else
356  {
357  /* Save the current Phase*/
358  TimingSettings.Phase24 = CurrentPhase;
359  }
360  }
361 
362  adi_CT_SetTimingsCfg(hDevice,&TimingSettings);
363 
364  }
365 
366  *Phase24Result = MinNoisePhase;
367  TimingSettings.Phase24 = MinNoisePhase;
368  adi_CT_SetTimingsCfg(hDevice,&TimingSettings);
369 
370 
371 
372  return ADI_CT_RESULT_SUCCESS;
373 }

393 ADI_CT_RESULT_TYPE adi_CT_NoiseMeasurement(ADI_CT_DEV_HANDLE const hDevice,ADI_CT_STAGE_NUMBER_TYPE StageNumber,uint32_t *variance,uint8_t MeasureNumber)
394 
395 {
396 
397  #define MEASURE_NUMBER 8
398  #define FIRST_NOISE_STAGE 6
399 
400 
401  ADI_STAGE_TYPE StageCfgType;
402  ADI_CT_INPUT_TYPE CDCInput;
403 
404  uint16_t NoiseStages = ADI_CT_STAGE6_BIT | ADI_CT_STAGE7_BIT | ADI_CT_STAGE8_BIT | ADI_CT_STAGE9_BIT | ADI_CT_STAGE10_BIT | ADI_CT_STAGE11_BIT | ADI_CT_STAGE12_BIT | ADI_CT_STAGE13_BIT;
405  uint16_t SignalStages = ADI_CT_STAGE0_BIT | ADI_CT_STAGE1_BIT | ADI_CT_STAGE2_BIT | ADI_CT_STAGE3_BIT | ADI_CT_STAGE4_BIT | ADI_CT_STAGE5_BIT;
406 
407  uint16_t NoiseValues[MEASURE_NUMBER];
408  uint32_t mean;
409  uint16_t den = 0;
410  uint32_t variancevalue = 0;
411 
412  uint8_t measure_counter = 0;
413 
414  /*
415  Disable any IIR, Ranking, CAPDAC configuration
416  */
417 
418 
419  /* To store the Previous IIR Filter Status */
420  bool_t PreviousFilterStatus;
421 
422  /* Store the IIR Filter Status */
423  adi_CT_GetIIRFilterEnable(hDevice,&PreviousFilterStatus);
424 
425 
426  ADI_CT_AVERAGE_TYPE PreviousAverageSettings;
427  ADI_CT_AVERAGE_TYPE AverageSettings;
428 
429  /* Average Settings */
430  adi_CT_GetAverageCfg(hDevice,&PreviousAverageSettings);
431 
432  AverageSettings = PreviousAverageSettings;
433 
434  AverageSettings.RankFilt = ADI_CT_RANKING_NO_RANKING;
435 
436  /* Average Settings */
437  adi_CT_SetAverageCfg(hDevice,&AverageSettings);
438 
439  /* To store the Previous Captouch Configuration */
440  uint8_t PreviousCaptouchInterrupts = pADI_CT->CT_IEN;
441 
442  /* To store the Previous IIR Filter Status */
443  bool_t PreviousSelfTimerStatus;
444 
445  /* Store the Self Timer Status */
446  adi_CT_GetSelfTimerEnable(hDevice,&PreviousSelfTimerStatus);
447 
448 
449  /* Disable IIR Filter */
450  adi_CT_SetIIRFilterEnable(hDevice,false);
451 
452  /* Get the Stage configuration to configure the stages to measure the noise using the same configuration */
453  adi_CT_GetStageGenericCfg(hDevice,StageNumber,&CDCInput,&StageCfgType);
454 
455  /* Enable the noise measurement stages and disable the other stages.*/
456  adi_CT_SetStagesEnable(hDevice,SignalStages,true);
457  adi_CT_SetStagesEnable(hDevice,NoiseStages,true);
458 
459  for(uint8_t i = FIRST_NOISE_STAGE ; i < FIRST_NOISE_STAGE + MEASURE_NUMBER ; i++)
460  {
461 
462  /* Configure the Stage */
463  adi_CT_SetStageGenericCfg(hDevice,(ADI_CT_STAGE_NUMBER_TYPE)(i),CDCInput,&StageCfgType);
464 
465  adi_CT_SetStageCAPDACEnable(hDevice,(ADI_CT_STAGE_NUMBER_TYPE)(i),false);
466 
467  /* Enable the Frequency Hopping Algorithm */
468  adi_CT_SetStageFreqHoppingEnable(hDevice,(ADI_CT_STAGE_NUMBER_TYPE)(i),true);
469 
470  }
471 
472  /* Enable the Conversion Complete Interrupt */
473  adi_CT_SetInterruptsEnable(hDevice,ADI_CT_INTERRUPT_CONV_SET_COMPLETE,true);
474 
475  /* Power Up the Captouch */
476  adi_CT_SetPowerStatus(hDevice,true);
477 
478  while(measure_counter < MeasureNumber)
479 
480  {
481 
482  /* Start a single conversion sequence and wait until it finishes (detected by CONV_SET_COMPLETE Interrupt) */
483  adi_CT_SetStartSingleSequence(hDevice);
484 
485  while (pADI_CT->CT_CDC_PWR & BITM_CT_CT_CDC_PWR_SW_STATUS);
486 
487 
488  /* Read the CDC Value and calculate the mean value*/
489  mean = 0;
490  for(uint8_t i = 0; i < MEASURE_NUMBER ; i++)
491  {
492  adi_CT_GetCDCValue(hDevice,(ADI_CT_STAGE_NUMBER_TYPE)(FIRST_NOISE_STAGE + i),&NoiseValues[i]);
493  mean += NoiseValues[i];
494  }
495  mean = mean >> 3;
496 
497 
498  /* variance */
499  variancevalue = 0;
500  for(uint8_t n = 0; n < MEASURE_NUMBER; n++)
501  {
502  den = abs(NoiseValues[n] - mean);
503  variancevalue += (den * den);
504  }
505  variancevalue = variancevalue >> 3;
506 
507  if(measure_counter == 0)
508 
509  {
510  *variance = variancevalue;
511  }
512 
513  else
514 
515  {
516  *variance = (*variance + variancevalue);
517  }
518 
519  measure_counter++;
520 
521 
522  }
523 
524  *variance = *variance/MeasureNumber;
525 
526  /* Average Settings */
527  adi_CT_SetAverageCfg(hDevice,&PreviousAverageSettings);
528 
529 
530  /* Enable the Conversion Complete Interrupt */
531  adi_CT_SetInterruptsEnable(hDevice,ADI_CT_INTERRUPT_CONV_SET_COMPLETE,false);
532 
533  /* Enable the Conversion Complete Interrupt */
534  adi_CT_SetInterruptsEnable(hDevice,PreviousCaptouchInterrupts,true);
535 
536 
537 
538 
539  /* Store the IIR Filter Status */
540  adi_CT_SetIIRFilterEnable(hDevice,PreviousFilterStatus);
541 
542  /* Set the Previous Self Timer Status */
543  adi_CT_SetSelfTimerEnable(hDevice,PreviousSelfTimerStatus);
544 
545  adi_CT_SetStagesEnable(hDevice,SignalStages,true);
546  adi_CT_SetStagesEnable(hDevice,NoiseStages,false);
547 
548  return ADI_CT_RESULT_SUCCESS;
549 
550 }
