186 ADI_GPT_RESULT_TYPE adi_GPT_Init(ADI_GPT_DEV_ID_TYPE InstanceNum, ADI_GPT_HANDLE* phDevice)
187 {
188  ADI_GPT_HANDLE hDevice;
189 
190  /* store a bad handle in case of failure */
191  *phDevice = (ADI_GPT_HANDLE) NULL;
192 
193 #ifdef ADI_DEBUG
194  if (InstanceNum >= ADI_GPT_MAX_DEVID)
195  {
196  return ADI_GPT_ERR_BAD_INSTANCE;
197  }
198 #endif
199 
200  /* local pointer to instance data */
201  hDevice = &gGPT_DevData[InstanceNum];
202 
203 #ifdef ADI_DEBUG
204  if (hDevice->bInitialized)
205  {
206  return ADI_GPT_ERR_ALREADY_INITIALIZED;
207  }
208 #endif
209 
210  /* initialize device instance data and install handlers */
211  switch (InstanceNum) {
212  case ADI_GPT_DEVID_0:
213  hDevice->pTmr = pADI_TM0;
214  hDevice->TimerIntID = (IRQn_Type)ADI_INT_TIMER0;
215  ADI_INSTALL_HANDLER(ADI_INT_TIMER0, GP_Tmr0_Int_Handler);
216  break;
217 
218  case ADI_GPT_DEVID_1:
219  hDevice->pTmr = pADI_TM1;
220  hDevice->TimerIntID = (IRQn_Type)ADI_INT_TIMER1;
221  ADI_INSTALL_HANDLER(ADI_INT_TIMER1, GP_Tmr1_Int_Handler);
222  break;
223  case ADI_GPT_DEVID_2:
224  hDevice->pTmr = pADI_TM2;
225  hDevice->TimerIntID = (IRQn_Type)ADI_INT_TIMER2;
226  ADI_INSTALL_HANDLER(ADI_INT_TIMER2, GP_Tmr2_Int_Handler);
227  break;
228 #ifdef ADI_DEBUG
229  default: return ADI_GPT_ERR_BAD_INSTANCE;
230 #endif
231  }
232 
233  /* do static register initializations */
234  ADI_GPT_CONFIG const *pConfig = &GPTConfig[InstanceNum];
235 
236  /* Copy the register configuration values into the controller registers */
237  hDevice->pTmr->GPTCON = pConfig->GPTCON;
238  hDevice->pTmr->GPTLD = pConfig->GPTLD;
239  hDevice->pTmr->GPTALD = pConfig->GPTALD;
240  hDevice->pTmr->GPTPCON = pConfig->GPTPCON;
241  hDevice->pTmr->GPTPMAT = pConfig->GPT_MATCH_VAL;
242 
243  hDevice->bInitialized = true;
244 
245  /* Enable timer interrupt */
246  ADI_ENABLE_INT(hDevice->TimerIntID);
247 
248  /* Clear the callback function and callback parameters */
249  hDevice->pfCallback = NULL;
250  hDevice->pCBParam = NULL;
251 
252  adi_GPT_ResetTimer( hDevice );
253 
254  *phDevice = hDevice; /* set handle to initialized Timer device */
255 
256  return ADI_GPT_SUCCESS;
257 }

274 ADI_GPT_RESULT_TYPE adi_GPT_UnInit(ADI_GPT_HANDLE const hDevice)
275 {
276 ADI_GPT_RESULT_TYPE result;
277 
278 #ifdef ADI_DEBUG
279  if (ADI_GPT_INVALID_HANDLE(hDevice))
280  {
281  return ADI_GPT_ERR_BAD_DEV_HANDLE;
282  }
283 #endif
284 
285  /* disable the timer and stub out the device pointer */
286  result = adi_GPT_SetTimerEnable(hDevice, false);
287 
288  /* Disable timer interrupt */
289  ADI_DISABLE_INT(hDevice->TimerIntID);
290 
291  ADI_UNINSTALL_HANDLER(hDevice->TimerIntID);
292 
293  hDevice->pTmr = (ADI_TIMER_TypeDef*)NULL;
294 
295  /* mark it uninitialized */
296  hDevice->bInitialized = false;
297 
298  return result;
299 }

877 /* Common interrupt handler, which notifies events to the application */
878 static void CommonIntHandler(ADI_GPT_DEV_DATA_TYPE *pDevice)
879 {
880  /* Pointer to timer registers */
881  ADI_TIMER_TypeDef *pTmr = pDevice->pTmr;
882 
883  /* Status register value */
884  uint16_t IntStatus = pTmr->STA;
885 
886  /* Clear the interrupts */
887  pTmr->CLRI = TCLRI_TMOUT_CLR;
888 
889  /* if a callback is installed notify the events */
890  if(pDevice->pfCallback)
891  {
892  /* Check if timeout occured and notify of timeout event */
893  if(IntStatus & BITM_GPT_GPTSTA_TMOUT)
894  {
895  pDevice->pfCallback(pDevice->pCBParam, ADI_GPT_EVENT_TIMEOUT, NULL);
896  }
897 
898  /* Check if event is captured and notify of captured event */
899  if(IntStatus & BITM_GPT_GPTSTA_CAP)
900  {
901  pDevice->pfCallback(pDevice->pCBParam, ADI_GPT_EVENT_CAPTURED, NULL);
902  }
903  }
904 
905  return;
906 }
