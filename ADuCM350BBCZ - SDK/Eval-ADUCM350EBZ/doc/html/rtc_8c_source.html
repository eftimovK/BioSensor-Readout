161 ADI_RTC_RESULT_TYPE adi_RTC_Init(ADI_RTC_DEV_ID_TYPE InstanceNum, ADI_RTC_HANDLE* phDevice)
162 {
163  ADI_RTC_HANDLE hDevice;
164 
165  /* store a bad handle in case of failure */
166  *phDevice = (ADI_RTC_HANDLE) NULL;
167 
168 #ifdef ADI_DEBUG
169  if (InstanceNum >= ADI_RTC_MAX_DEVID)
170  {
171  return ADI_RTC_ERR_BAD_INSTANCE;
172  }
173 #endif
174 
175  /* local pointer to instance data */
176  hDevice = &gRTC_DevData[InstanceNum];
177 
178 #ifdef ADI_DEBUG
179  if (hDevice->bInitialized)
180  {
181  return ADI_RTC_ERR_ALREADY_INITIALIZED;
182  }
183 #endif
184 
185  /* make sure RTC has a clock so it can respond */
186  /* do this even before failing out on RTCFAIL, */
187  /* or application efforts to clear it with */
188  /* adi_RTC_ClearFailSafe will fail */
189  SystemClockSourcePowerUp (ADI_SYS_CLOCK_SOURCE_LFXTAL, ADI_SYS_CLOCK_TIMEOUT_USE_DEFAULT);
190 
191  /* fail on corrupt clock */
192  if (ADI_RTC_FAILSAFE)
193  return ADI_RTC_ERR_CLOCK_FAILSAFE;
194 
195  /* initialize device data entries */
196  hDevice->pRtc = pADI_RTC;
197  hDevice->bSafe = true;
198  hDevice->cbFunc = NULL;
199  hDevice->cbWatch = (ADI_RTC_INT_SOURCE_TYPE) 0;
200  hDevice->IRQn = RTC_IRQn;
201 
202  /* disable all interrupt sources */
203  while (hDevice->pRtc->RTCSR1 & ADI_RTC_WRITE_STATUS_CONTROL)
204  ;
205  hDevice->pRtc->RTCCR &= ~ADI_RTC_INT_ENA_MASK;
206 
207  /* clear any pending interrupts */
208  while (hDevice->pRtc->RTCSR1 & ADI_RTC_WRITE_STATUS_STATUS0)
209  ;
210  hDevice->pRtc->RTCSR0 = ADI_RTC_INT_SOURCE_MASK;
211 
212  /* install interrupt handler (see abstraction macro for OS in adi_rtos.h) */
213  ADI_INSTALL_HANDLER(hDevice->IRQn, RTC_Int_Handler);
214 
215  /* Enable RTC interrupts in NVIC */
216  ADI_ENABLE_INT(hDevice->IRQn);
217 
218  /* mark driver initialized */
219  hDevice->bInitialized = true;
220 
221  /* store handle at application handle pointer */
222  *phDevice = hDevice;
223 
224  return ADI_RTC_SUCCESS; // initialized
225 }

677 ADI_RTC_RESULT_TYPE adi_RTC_GetAlarm (ADI_RTC_HANDLE hDevice, uint32_t *pAlarm)
678 {
679  uint32_t result;
680 
681 #ifdef ADI_DEBUG
682  if (ADI_RTC_INVALID_HANDLE(hDevice))
683  return ADI_RTC_ERR_BAD_DEV_HANDLE;
684

686  return ADI_RTC_ERR_NOT_INITIALIZED;
687

690 
691  if (!pAlarm)
692  return ADI_RTC_ERR_PARAM_OUT_OF_RANGE;
693 #endif
694 
695  /* disable interrupts during paired read */
696  ADI_DISABLE_INT(hDevice->IRQn);
697  result = hDevice->pRtc->RTCALM1 << 16;
698  result |= hDevice->pRtc->RTCALM0;
699  ADI_ENABLE_INT(hDevice->IRQn);
700 
701  *pAlarm = result;
702 
703  return ADI_RTC_SUCCESS;

1302 ADI_RTC_RESULT_TYPE adi_RTC_SetCount (ADI_RTC_HANDLE hDevice, uint32_t Count)
1303 {
1304 #ifdef ADI_DEBUG
1305  if (ADI_RTC_INVALID_HANDLE(hDevice))
1306  return ADI_RTC_ERR_BAD_DEV_HANDLE;
1307 
1308  if (!hDevice->bInitialized)
1309  return ADI_RTC_ERR_NOT_INITIALIZED;
1310 
1311  if (ADI_RTC_FAILSAFE)
1312  return ADI_RTC_ERR_CLOCK_FAILSAFE;
1313 #endif
1314 
1315  /* pause on pending writes to both CNTs to avoid data loss */
1316  if (hDevice->bSafe)
1317  while (hDevice->pRtc->RTCSR1 & (ADI_RTC_WRITE_STATUS_COUNT0 | ADI_RTC_WRITE_STATUS_COUNT1))
1318  ;
1319 
1320  ADI_ENTER_CRITICAL_REGION();
1321  /* RTC hardware insures paired write, so no need to disable interrupts */
1322  hDevice->pRtc->RTCCNT0 = Count & 0xffff;
1323  hDevice->pRtc->RTCCNT1 = (Count >> 16) & 0xffff;
1324  ADI_EXIT_CRITICAL_REGION();
1325 
1326  /* clear RTCFAIL interrupt bit if its set */
1327  if (pADI_RTC->RTCSR0 & ADI_RTC_INT_SOURCE_FAIL) {

1745 ADI_INT_HANDLER (RTC_Int_Handler)
1746 {
1747  ADI_RTC_HANDLE pDev = &gRTC_DevData[ADI_RTC_DEVID_0]; // so far, there is only one RTC, so this is safe
1748  uint16_t sources, enables;
1749  ADI_RTC_INT_SOURCE_TYPE fired;
1750 
1751  /* determin qualified interrupt source(s) */
1752 
1753  sources = pDev->pRtc->RTCSR0 & ADI_RTC_INT_SOURCE_MASK;
1754  enables = pDev->pRtc->RTCCR & ADI_RTC_INT_ENA_MASK;
1755 
1756  /* start with unmaskable RTCFAIL bit */
1757  fired = (ADI_RTC_INT_SOURCE_TYPE) (sources & ADI_RTC_INT_SOURCE_FAIL);
1758 
1759  /* need to test each interrupt source and whether it is enabled before notifying */
1760  /* because each source is latched regardless of whether it is enabled or not :-( */
1761  if ((ADI_RTC_INT_SOURCE_ALARM & sources)
1762  && (ADI_RTC_INT_ENA_ALARM & enables))
1763  fired |= ADI_RTC_INT_SOURCE_ALARM;
1764 
1765  if ((ADI_RTC_INT_SOURCE_LCD_UPDATE & sources)
1766  && (ADI_RTC_INT_ENA_LCD_UPDATE & enables))
1767  fired |= ADI_RTC_INT_SOURCE_LCD_UPDATE;
1768 
1769  if ((ADI_RTC_INT_SOURCE_ISO_DONE & sources)
1770  && (ADI_RTC_INT_ENA_ISO_DONE & enables))
1771  fired |= ADI_RTC_INT_SOURCE_ISO_DONE;
1772 
1773  if ((ADI_RTC_INT_SOURCE_WRITE_PENDERR & sources)
1774  && (ADI_RTC_INT_ENA_WRITE_PENDERR & enables))
1775  fired |= ADI_RTC_INT_SOURCE_WRITE_PENDERR;
1776 
1777  if ((ADI_RTC_INT_SOURCE_WRITE_SYNC & sources)
1778  && (ADI_RTC_INT_ENA_WRITE_SYNC & enables))
1779  fired |= ADI_RTC_INT_SOURCE_WRITE_SYNC;
1780 
1781  if ((ADI_RTC_INT_SOURCE_WRITE_PEND & sources)
1782  && (ADI_RTC_INT_ENA_WRITE_PEND & enables))
1783  fired |= ADI_RTC_INT_SOURCE_WRITE_PEND;
1784 
1785  /* 'fired' now represents interrupt bits that are both set and enabled... */
1786 
1787  /* if we have a callback... */
1788  if (pDev->bInitialized && pDev->cbFunc) {
1789 
1790  /* forward to the user if he is watching this interrupt */
1791  /* pass the "fired" value as the event. argument param is not used */
1792  if (pDev->cbWatch & fired)
1793  pDev->cbFunc (pDev, fired, NULL);
1794  }
1795 
1796  /* clear the qualified interrupt(s) */
1797  adi_RTC_ClearInterruptStatus(pDev, fired);
1798 }
