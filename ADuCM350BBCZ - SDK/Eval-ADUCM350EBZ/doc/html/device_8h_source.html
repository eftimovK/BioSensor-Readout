288 /*
289  * Bit-Banding Macros for atomic RAM/MMR bit operations.
290  * Compute the incoming RAM/MMR address and bit position
291  * bit-banded address alias and set/clear the bit.
292  */
293 #define RAM_ADR_BASE 0x20000000
294 #define RAM_BB_ALIAS 0x22000000
295 #define MMR_ADR_BASE 0x40000000
296 #define MMR_BB_ALIAS 0x42000000
297 #define MMR_BB_SET(a,b) ( *(volatile uint32_t *)(MMR_BB_ALIAS | (((uint32_t)a - MMR_ADR_BASE) << 5) | (b << 2)) = 1 )
298 #define MMR_BB_CLR(a,b) ( *(volatile uint32_t *)(MMR_BB_ALIAS | (((uint32_t)a - MMR_ADR_BASE) << 5) | (b << 2)) = 0 )
299 #define RAM_BB_SET(a,b) ( *(volatile uint32_t *)(RAM_BB_ALIAS | (((uint32_t)a - RAM_ADR_BASE) << 5) | (b << 2)) = 1 )
300 #define RAM_BB_CLR(a,b) ( *(volatile uint32_t *)(RAM_BB_ALIAS | (((uint32_t)a - RAM_ADR_BASE) << 5) | (b << 2)) = 0 )
