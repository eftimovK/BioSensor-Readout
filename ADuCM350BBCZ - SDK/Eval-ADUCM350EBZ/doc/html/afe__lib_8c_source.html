78 ADI_AFE_RESULT_TYPE adi_AFE_AuxChanCal(ADI_AFE_DEV_HANDLE const hDevice) {
79  ADI_AFE_RESULT_TYPE result = ADI_AFE_SUCCESS;;
80  uint16_t rxBuffer[2];
81  int32_t offset;
82  uint32_t gain;
83 
84 #ifdef ADI_DEBUG
85  if (adi_AFE_InvalidHandle(hDevice)) {
86  return ADI_AFE_ERR_BAD_DEV_HANDLE;
87  }
88 
89  if (adi_AFE_HandleNotInitialized(hDevice)) {
90  return ADI_AFE_ERR_NOT_INITIALIZED;
91  }
92 
93  if (pADI_SYSCLK->CLKCON5 & BITM_SYSCLK_CLKCON5_ACLKOFF) {
94  return ADI_AFE_ERR_ACLKOFF;
95  }
96 
97  if (16000000 != SystemGetClockFrequency(ADI_SYS_CLOCK_ACLK)) {
98  return ADI_AFE_ERR_WRONG_ACLK_FREQUENCY;
99  }
100 #endif
101 
102  /* Run the sequence to perform 2 measurements and store them in rxBuffer through DMA */
103  result = adi_AFE_RunSequence(hDevice, seq_afe_auxchancal, rxBuffer, 2);
104 
105  if (!result) {
106 
107  /* Performing 2-point calibration */
108  /* gain = 2^15 / (rxBuffer[1] - rxBuffer[0]) */
109  /* offset = -20025 / gain - (rxBuffer[0] - 32768) */
110  /* -20025 is the expected result from the first measurement, as a signed number */
111 
112  /* Second measurement always yields a higher code than the first measurement */
113  /* gain is an unsigned number */
114  gain = (32768 * 16384 * 2) / (rxBuffer[1] - rxBuffer[0]);
115  /* Rounding */
116  gain = (gain + 1) >> 1;
117 
118  /* Calculate the offset with 0.25LSB resolution */
119  offset = -20025 * (rxBuffer[1] - rxBuffer[0]);
120  /* 0.125LSB resolution */
121  offset = offset >> 12;
122  /* Round to 0.25LSB resolution */
123  offset = (offset + 1) >> 1;
124  offset -= (rxBuffer[0] << 2);
125  offset += (32768 << 2);
126 
127  /* Write gain and offset values to the calibration registers */
128  if (ADI_AFE_SUCCESS != (result = adi_AFE_WriteCalibrationRegister(hDevice, ADI_AFE_CAL_REG_ADC_OFFSET_AUX, offset)))
129  {
130  return result;
131  }
132  if (ADI_AFE_SUCCESS != (result = adi_AFE_WriteCalibrationRegister(hDevice, ADI_AFE_CAL_REG_ADC_GAIN_AUX, gain)))
133  {
134  return result;
135  }
136  }
137 
138  return result;
139 }

441 ADI_AFE_RESULT_TYPE adi_AFE_TiaChanCal(ADI_AFE_DEV_HANDLE const hDevice) {
442  ADI_AFE_RESULT_TYPE result = ADI_AFE_SUCCESS;
443  uint32_t seq_local[9];
444  uint32_t rcal;
445  uint32_t rtia;
446  uint32_t tiaChanDacCode;
447  int32_t offset_code;
448  uint32_t gain_code;
449  int32_t expected_code;
450  int32_t channel_offset;
451  int32_t channel_gain;
452  int32_t tmp;
453 
454 #ifdef ADI_DEBUG
455  if (adi_AFE_InvalidHandle(hDevice)) {
456  return ADI_AFE_ERR_BAD_DEV_HANDLE;
457  }
458 
459  if (adi_AFE_HandleNotInitialized(hDevice)) {
460  return ADI_AFE_ERR_NOT_INITIALIZED;
461  }
462 
463  if (pADI_SYSCLK->CLKCON5 & BITM_SYSCLK_CLKCON5_ACLKOFF) {
464  return ADI_AFE_ERR_ACLKOFF;
465  }
466 
467  if (16000000 != SystemGetClockFrequency(ADI_SYS_CLOCK_ACLK)) {
468  return ADI_AFE_ERR_WRONG_ACLK_FREQUENCY;
469  }
470 #endif
471 
472  /* Get RCAL value */
473  if (ADI_AFE_SUCCESS != (result = adi_AFE_GetRcal(hDevice, &rcal)))
474  {
475  return result;
476  }
477 
478  /* Get RTIA value */
479  if (ADI_AFE_SUCCESS != (result = adi_AFE_GetRtia(hDevice, &rtia)))
480  {
481  return result;
482  }
483 
484  /* Get TIA channel DAC code */
485  if (ADI_AFE_SUCCESS != (result = adi_AFE_TiaChanDacCode(hDevice, &tiaChanDacCode)))
486  {
487  return result;
488  }
489 
490  /* Start the calibration routine with default gain and offset values */
491  adi_AFE_WriteCalibrationRegister(hDevice, ADI_AFE_CAL_REG_ADC_GAIN_TIA, (uint32_t)0x4000);
492  adi_AFE_WriteCalibrationRegister(hDevice, ADI_AFE_CAL_REG_ADC_OFFSET_TIA, (uint32_t)0x000);
493 
494  /* Step 1 */
495  /* Measure 0V (P = Vbias, N = Vbias) */
496 
497  /* Copy the seq_afe_tiachancal1 sequence to a local variable so that it can be updated */
498  for (uint32_t i = 0; i < 9; i++)
499  {
500  seq_local[i] = seq_afe_tiachancal1[i];
501  }
502 
503  seq_local[2] = 0xAA000000 + tiaChanDacCode;
504 
505  /* Recalculate CRC in software because we changed the original sequence */
506  adi_AFE_EnableSoftwareCRC(hDevice, true);
507 
508  result = adi_AFE_RunSequence(hDevice, seq_local, NULL, 0);
509 
510  /* Restore to using default CRC stored with the sequence */
511  adi_AFE_EnableSoftwareCRC(hDevice, false);
512 
513  if (!result) {
514 
515  /* Read measurement result, convert to signed format and invert the sign */
516  offset_code = 32768 - pADI_AFE->AFE_SUPPLY_LPF_RESULT;
517  /* Align to 0.25LSB resolution */
518  offset_code = offset_code << 2;
519 
520  /* Write offset to the calibration register */
521  adi_AFE_WriteCalibrationRegister(hDevice, ADI_AFE_CAL_REG_ADC_OFFSET_TIA, offset_code);
522 
523  /* Step 2 */
524  /* Measure 0.7V (P = Vref, N = Vbias) */
525  result = adi_AFE_RunSequence(hDevice, seq_afe_tiachancal2, NULL, 0);
526  }
527 
528  if (!result) {
529 
530  /* Read measurement result and convert to a signed number */
531  gain_code = pADI_AFE->AFE_SUPPLY_LPF_RESULT - 32768;
532  /* Gain code calculation: (0.7V * 1.5 / 3.6V) * 65535 */
533  gain_code = (16384 * 19114) / gain_code;
535 
536  /* Write gain to the calibration register */
537  adi_AFE_WriteCalibrationRegister(hDevice, ADI_AFE_CAL_REG_ADC_GAIN_TIA, gain_code);
538 
539  /* Step 3 */
540  /* Measure voltage across Rcal*/
541  result = adi_AFE_RunSequence(hDevice, seq_afe_tiachancal3, NULL, 0);
542  }
543 
544  if (!result) {
545 
546  /* Read measurement result, convert to signed format and store for later usage */
547  expected_code = pADI_AFE->AFE_SUPPLY_LPF_RESULT - 32768;
548 
549  /* Step 4 */
550  /* Measure TIA channel offset (P = IVAO, N = IVAIM) */
551  result = adi_AFE_RunSequence(hDevice, seq_afe_tiachancal4, NULL, 0);
552  }
553 
554  if (!result) {
555 
556  channel_offset = 32768 - pADI_AFE->AFE_SUPPLY_LPF_RESULT;
557  channel_offset = channel_offset << 2;
558  /* Add normalized offset to TIA offset */
560  /* Truncation vs. rounding to be revised later */
561  channel_offset += (offset_code * 16384) / gain_code;
562 
563  /* Write offset to the calibration register */
564  adi_AFE_WriteCalibrationRegister(hDevice, ADI_AFE_CAL_REG_ADC_OFFSET_TIA, channel_offset);
565 
566  /* Step 5 */
567  /* Measure TIA channel gain (P = IVAO, N = IVAIM) */
568  result = adi_AFE_RunSequence(hDevice, seq_afe_tiachancal5, NULL, 0);
569  }
570 
571  if (!result) {
572 
573  /* Read measurement result and convert to signed format */
574  /* Result from TIA is "inverted" (i.e. positive current gives negative voltage) */
575  channel_gain = 32768 - pADI_AFE->AFE_SUPPLY_LPF_RESULT;
576 
577  /* Current measured in step 3 */
578  /* channel_gain = ((16384 * (rtia / rcal)) * expected_code) / channel_gain */
579  tmp = (16384 * rtia) / rcal;
580  channel_gain = (tmp * expected_code) / channel_gain;
581 
582  /* Multiply by the previously calculated gain and scale */
583  /* Rounding to give 0.5 LSB error */
584  channel_gain = ((channel_gain * gain_code) + (1 << 13)) >> 14;
585 
586  /* Write gain to the calibration register */
587  adi_AFE_WriteCalibrationRegister(hDevice, ADI_AFE_CAL_REG_ADC_GAIN_TIA, (uint32_t)channel_gain);
588 
589  }
590 
591  return result;
592 }

613 ADI_AFE_RESULT_TYPE adi_AFE_ExciteChanCalAtten(ADI_AFE_DEV_HANDLE const hDevice) {
614  ADI_AFE_RESULT_TYPE result = ADI_AFE_SUCCESS;
615  uint16_t adcResult[2];
616  int32_t offset;
617 
618 #ifdef ADI_DEBUG
619  if (adi_AFE_InvalidHandle(hDevice)) {
620  return ADI_AFE_ERR_BAD_DEV_HANDLE;
621  }
622 
623  if (adi_AFE_HandleNotInitialized(hDevice)) {
624  return ADI_AFE_ERR_NOT_INITIALIZED;
625  }
626 
627  if (pADI_SYSCLK->CLKCON5 & BITM_SYSCLK_CLKCON5_ACLKOFF) {
628  return ADI_AFE_ERR_ACLKOFF;
629  }
630 
631  if (16000000 != SystemGetClockFrequency(ADI_SYS_CLOCK_ACLK)) {
632  return ADI_AFE_ERR_WRONG_ACLK_FREQUENCY;
633  }
634 #endif
635 
636  /* Write 0x500 to DAC (0x800 + (0xA00 >> 1)) */
637  adi_AFE_WriteCalibrationRegister(hDevice, ADI_AFE_CAL_REG_DAC_OFFSET_ATTEN, 0x00000A00);
638 
639  result = adi_AFE_RunSequence(hDevice, seq_afe_excitechancalatten1, NULL, 0);
640 
641  if (result) {
642  return result;
643  }
644 
645  adcResult[0] = pADI_AFE->AFE_SUPPLY_LPF_RESULT;
646 
647  /* Write 0xB00 to DAC (0x800 + (0x600 >> 1)) */
648  adi_AFE_WriteCalibrationRegister(hDevice, ADI_AFE_CAL_REG_DAC_OFFSET_ATTEN, 0x00000600);
649 
650  result = adi_AFE_RunSequence(hDevice, seq_afe_excitechancalatten2, NULL, 0);
651 
652  if (result) {
653  return result;
654  }
655 
656  adcResult[1] = pADI_AFE->AFE_SUPPLY_LPF_RESULT;
657 
658 /*
659  * Interpolate
660  *
661  * Using formula: (x - x0) / (x1 - x0) = (y - y0) / (y1 - y0)
662  * where:
663  * x = 0x800 + offset
664  * x0 = 0x500
665  * x1 = 0xB00
666  * y = 0x8000
667  * y0 = adcResult[0]
668  * y1 = adcResult[1]
669  *
670  * x = x0 + (x1 - x0) * (y - y0) / (y1 - y0)
671  * = 0x500 + 0x600 * (0x8000 - adcResult[0]) / (adcResult[1] - adcResult[0])
672  *
673  * offset = 0x600 * (0x8000 - adcResult[0]) / (adcResult[1] - adcResult[0]) - 0x300
674  *
675  */
676 
677  offset = 0x00008000 - (int32_t)adcResult[0];
678  offset *= 0x00000600;
679  /* Increase resolution to 0.25 LSB */
680  offset = offset << 2;
681  offset /= (adcResult[1] - adcResult[0]);
682  offset -= (0x00000300 << 2);
683  /* Round to 0.5 LSB */
684  offset = (offset + 1) >> 1;
685 
686  /* Write measured offset value to the calibration register */
687  adi_AFE_WriteCalibrationRegister(hDevice, ADI_AFE_CAL_REG_DAC_OFFSET_ATTEN, offset);
688 
689  return result;
690 }

775  */
776 
777  tmp1 = exChanNoAttenDeltaDacCode * (0x8000 - adcResult[0]);
778  tmp2 = adcResult[1] - adcResult[0];
779 
780  /* Adding 0.25 LSB precision */
781  offset = (tmp1 << 2) / tmp2;
782  /* Rounding to 0.5 LSB precision */
783  offset = (offset + 1) >> 1;
784  /* Subtract the delta */
785  offset = offset - exChanNoAttenDeltaDacCode;
786 
787  /* Write measured offset value to the calibration register */
788  adi_AFE_WriteCalibrationRegister(hDevice, ADI_AFE_CAL_REG_DAC_OFFSET_UNITY, offset);
789 
790  return result;
791 }
