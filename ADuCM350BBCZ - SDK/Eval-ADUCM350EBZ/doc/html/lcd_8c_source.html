169 ADI_LCD_RESULT_TYPE adi_LCD_Init(ADI_LCD_DEV_ID_TYPE const devID, ADI_LCD_DEV_HANDLE* const phDevice)
170 {
171  ADI_LCD_DEV_HANDLE hDevice;
172 
173 #ifdef ADI_DEBUG
174  if (devID >= ADI_LCD_MAX_DEVID)
175  return ADI_LCD_ERR_BAD_DEVICE_INDEX;
176 
177  if (ADI_LCD_STATE_UNINITIALIZED != LCD_DevData[devID].initState)
178  return ADI_LCD_ERR_ALREADY_INITIALIZED;
179 #endif
180 
181  /* set local device handle */
182  hDevice = &LCD_DevData[devID];
183 
184  /* store a bad handle in case of failure */
185  *phDevice = NULL;
186 
187  /* Reset the LCD data registers */
188  hDevice->pLCD->LCDCON |= BITM_LCD_LCDCON_LCDRST;
189 
191  ADI_LCD_CFG_TYPE const *pLCDCfg = &gLCDCfg[devID];
192 
193  /* Initialize the device based on the given configuration parameters */
194  hDevice->pLCD->LCDCON = pLCDCfg->LCDCON;
195  hDevice->pLCD->LCDBLINK = pLCDCfg->LCDBLINK;
196  hDevice->pLCD->LCDCONTRAST = pLCDCfg->LCDCONTRAST;
197 
198  /* Initalize device variables */
199  hDevice->pfCallback = NULL;
200  hDevice->pCBParam = NULL;
201 
202  if(devID == 0)
203  {
204  /* Install the LCD interrupt handlers */
205  ADI_INSTALL_HANDLER(ADI_INT_LCD, LCD_Int_Handler);
206  }
207  /*
208  Add other instances here.
209  */
210 
211  /* init successful */
212  hDevice->initState = ADI_LCD_STATE_INITIALIZED;
213 
214  /* store the "device" handle last */
215  *phDevice = hDevice;
216 
217  return ADI_LCD_SUCCESS;
218 }

299 ADI_LCD_RESULT_TYPE adi_LCD_SetBlinkMode(ADI_LCD_DEV_HANDLE const hDevice, ADI_LCD_BLINK_MODE_TYPE const BlinkMode)
300 {
301  /* LCD MMR register value */
302  uint16_t RegValue;
303 
304  #ifdef ADI_DEBUG
305  if (ADI_LCD_SUCCESS != ValidateDeviceHandle(hDevice))
306  return ADI_LCD_ERR_BAD_DEVICE_HANDLE;
307 
308  if (ADI_LCD_STATE_UNINITIALIZED == hDevice->initState)
309  return ADI_LCD_ERR_NOT_INITIALIZED;
310 #endif
311 
312  /* Enable or disable blinking based on the chosen mode. If hardware or software mode is chosen enable blinking. */
313  RegValue = hDevice->pLCD->LCDCON;
314  RegValue &= (~BITM_LCD_LCDCON_BLINKEN);
315  RegValue |= (BlinkMode == ADI_LCD_BLINK_MODE_DISABLE) ? 0 : BITM_LCD_LCDCON_BLINKEN;
316  hDevice->pLCD->LCDCON = RegValue;
317 
318  /* Set the chosen hardware of software mode in blink control register. */
319  if(ADI_LCD_BLINK_MODE_DISABLE != BlinkMode)
320  {
321  RegValue = hDevice->pLCD->LCDBLINK;
322  RegValue &= (~BITM_LCD_LCDBLINK_BLKMOD);
323  RegValue |= BlinkMode;
324  hDevice->pLCD->LCDBLINK = RegValue;
325  }
326 
327  return ADI_LCD_SUCCESS;
328 }

822 {
823  /* LCD MMR register value */
824  uint16_t RegValue;
825 
826  #ifdef ADI_DEBUG
827  if (ADI_LCD_SUCCESS != ValidateDeviceHandle(hDevice))
828  return ADI_LCD_ERR_BAD_DEVICE_HANDLE;
829 
830  if (ADI_LCD_STATE_UNINITIALIZED == hDevice->initState)
831  return ADI_LCD_ERR_NOT_INITIALIZED;
832 #endif
833 
834  /* Enable or disable the auto switch. */
835  RegValue = hDevice->pLCD->LCDCONTRAST;
836  RegValue &= ~(BITM_LCD_LCDCONTRAST_CP_PD | BITM_LCD_LCDCONTRAST_CP_EN);
837 
838  if(bFlag == true)
839  {
840  /* To enable CP_PD bit should be set to 0 and CP_EN bit should be set to 1 */
841  RegValue |= (1 << BITP_LCD_LCDCONTRAST_CP_EN);
842  }
843  else
844  {
845  /* To disable CP_PD bit should be set to 1 and CP_EN bit should be set to 0 */
846  RegValue |= (1 << BITP_LCD_LCDCONTRAST_CP_PD);
847  }
848 
849  hDevice->pLCD->LCDCONTRAST = RegValue;
850 
851  return ADI_LCD_SUCCESS;
852 }
