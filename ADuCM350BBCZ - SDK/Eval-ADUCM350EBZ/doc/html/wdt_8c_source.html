123 ADI_WDT_RESULT_TYPE adi_WDT_Init(ADI_WDT_DEV_ID_TYPE const DevID, ADI_WDT_DEV_HANDLE * const phDevice)
124 {
125  ADI_WDT_DEV_HANDLE hDevice = &WDT_DevData[DevID]; /* local handle to device data structure */
126 
127  /* store a bad handle in case of failure */
128  *phDevice = (ADI_WDT_DEV_HANDLE) NULL;
129  hDevice->bBusSyncWait = true;
130  hDevice->InitState = ADI_WDT_STATE_UNINITIALIZED;
131 #ifdef ADI_DEBUG
132  if (DevID >= ADI_WDT_MAX_DEVID)
133  return ADI_WDT_ERR_BAD_DEV_ID;
134 
135  if (ADI_WDT_STATE_UNINITIALIZED != hDevice->InitState)
136  return ADI_WDT_ERR_ALREADY_INITIALIZED;
137 #endif
138 
139  /* if WDT not locked, and disabled (i.e. timer is not active) ... */
140  if (!(pADI_WDT->T3STA & T3STA_LOCK) && !(pADI_WDT->T3CON & T3CON_ENABLE)) {
141 
142  /* WDT is not locked and not enabled, so proceed to configure it... */
143 
144  /* assert pre-configured static data... */
145  ADI_WDT_STATIC_INIT_TYPE const* pInitData = &gWdtStaticInitData[DevID];
146 
147  pADI_WDT->T3LD = pInitData->wdtLoadRegister;
148  pADI_WDT->T3CON = pInitData->wdtControlRegister;
149  pADI_WDT->T3RCR = pInitData->wdtResetRegister;
150 
151  } /* end of unlocked & disabled WDT init block */
152 
153  /* else, WDT is either:
154  a) locked (i.e. its already configured and running), in which case
155  we can't do anything (the user can still use these APIs to reset
156  the timer and perform status checks), OR,
157  b) not locked, but is enabled (i.e. the timer is running), in which
158  case we also can't make any changes until the user decides to disable
159  and re-configure the timer
160  bottom line: no further WDT configurations to perform at this time...
161  */
162 
163  /* install interrupt handler (see abstraction macro for OS in adi_rtos.h) */
164  hDevice->IRQn = WDT_IRQn;
165  ADI_INSTALL_HANDLER(hDevice->IRQn, WDog_Tmr_Int_Handler);
166 
167  /* Enable WDT interrupts in NVIC */
168  ADI_ENABLE_INT(hDevice->IRQn);
169 
170  hDevice->InitState = ADI_WDT_STATE_INITIALIZED; /* mark device as initialized */
171  *phDevice = hDevice; /* write user handle */
172 
173  /* kick the dog on the way out, maximizing application time to configure... */
174  if (ADI_WDT_SUCCESS != adi_WDT_ResetTimer(hDevice))
175  return ADI_WDT_ERR_INITIALIZATION_FAIL;
176 
177  return ADI_WDT_SUCCESS;
178 }

193 ADI_WDT_RESULT_TYPE adi_WDT_UnInit(ADI_WDT_DEV_HANDLE const hDevice)
194 {
195 #ifdef ADI_DEBUG
196  /* ensure handle points to a valid watchdog device ... */
197  if (ADI_WDT_INVALID_HANDLE(hDevice))
198  return ADI_WDT_ERR_BAD_DEV_DATA;
199 
200  /* check that watchdog device is initialized ... */
201  if (ADI_WDT_STATE_INITIALIZED != hDevice->InitState)
202  return ADI_WDT_ERR_NOT_INITIALIZED;
203 #endif
204 
205  /* it makes no sense to "uninit" the timer if its locked
206  so this must indicate the failure */
207 #ifdef ADI_WDT_HW_ANOMALY_RESOLVED
208  if (pADI_WDT->T3STA & T3STA_LOCK)
209  return ADI_WDT_ERR_LOCKED;
210 #endif
211 
212  /* uninitialize */
213  ADI_DISABLE_INT(hDevice->IRQn);
214  ADI_UNINSTALL_HANDLER(hDevice->IRQn);
215  hDevice->IRQn = (IRQn_Type) 0;
216 
217  /* If timer not started, simply abandon "ownership" of the WDT */
218  hDevice->InitState = ADI_WDT_STATE_UNINITIALIZED; /* watchdog no longer in use */
219  return ADI_WDT_SUCCESS;
220 }

306 ADI_WDT_RESULT_TYPE adi_WDT_SetEnable(ADI_WDT_DEV_HANDLE const hDevice, const bool_t bEnable)
307 {
308 #ifdef ADI_DEBUG
309  /* ensure handle points to a valid watchdog device ... */
310  if (ADI_WDT_INVALID_HANDLE(hDevice))
311  return ADI_WDT_ERR_BAD_DEV_DATA;
312 
313  /* check that watchdog device is initialized ... */
314  if (ADI_WDT_STATE_INITIALIZED != hDevice->InitState)
315  return ADI_WDT_ERR_NOT_INITIALIZED;
316 #endif
317 
318  /* ensure that the watchdog is not locked out ... */
319 #ifdef ADI_WDT_HW_ANOMALY_RESOLVED
320  if (pADI_WDT->T3STA & T3STA_LOCK)
321  return ADI_WDT_ERR_LOCKED;
322 #endif
323 
324  /* wait on WDT sync if requested */
325  if (hDevice->bBusSyncWait) {
326  while (WDT_SYNC_PENDING) ;
327  } else {
328  if (WDT_SYNC_PENDING)
329  return ADI_WDT_SYNC_IN_PROGRESS;
330  }
331 
332  ADI_ENTER_CRITICAL_REGION();
333  if (bEnable) {
334  /* set periodic mode and enable bits */
335  pADI_WDT->T3CON |= (T3CON_MOD_PERIODIC | T3CON_ENABLE);
336  } else {
337  /* Clear the enable bit to disable the timer */
338  pADI_WDT->T3CON &= (~T3CON_ENABLE);
339  }
340  ADI_EXIT_CRITICAL_REGION();
341 
342  /* NOTE: This code assumes that the user does not need to know when the enable or disable
343  actually occurs as a result of the write to T3CON, hence this does not do a bus-sync
344  check after the write. Once enabled, it cannot be disabled, or reconfigured, so its
345  irrelevant and they do not need to wait for the sync to occur */
346 
347  return ADI_WDT_SUCCESS;
348 }

389 ADI_WDT_RESULT_TYPE adi_WDT_SetPrescale(ADI_WDT_DEV_HANDLE const hDevice, const ADI_WDT_PRESCALE_TYPE Prescale)
390 {
391  uint16_t tcon;
392 
393 #ifdef ADI_DEBUG
394  /* ensure handle points to a valid watchdog device ... */
395  if (ADI_WDT_INVALID_HANDLE(hDevice))
396  return ADI_WDT_ERR_BAD_DEV_DATA;
397 
398  /* check that watchdog device is initialized ... */
399  if (ADI_WDT_STATE_INITIALIZED != hDevice->InitState)
400  return ADI_WDT_ERR_NOT_INITIALIZED;
401 #endif
402 
403  /* ensure that the watchdog is not locked out ... */
404 #ifdef ADI_WDT_HW_ANOMALY_RESOLVED
405  if (pADI_WDT->T3STA & T3STA_LOCK)
406  return ADI_WDT_ERR_LOCKED;
407 #endif
408 
409  switch (Prescale) {
410 
411  case ADI_WDT_PRESCALE_1:
412  case ADI_WDT_PRESCALE_16:
413  case ADI_WDT_PRESCALE_256:
414  case ADI_WDT_PRESCALE_4096:
415 
416  /* wait on WDT sync if requested */
417  if (hDevice->bBusSyncWait) {
418  while (WDT_SYNC_PENDING) ;
419  } else {
420  if (WDT_SYNC_PENDING)
421  return ADI_WDT_SYNC_IN_PROGRESS;
422  }
423 
424  /* critical section */
425  ADI_ENTER_CRITICAL_REGION();
426  tcon = pADI_WDT->T3CON; /* temp copy */
427  tcon &= ~(T3CON_PRE_MSK); /* make a hole */
428  tcon |= Prescale; /* rollin new prescale */
429  pADI_WDT->T3CON = tcon; /* write it back */
430  ADI_EXIT_CRITICAL_REGION();
431 
432  break;
433 
434  default:
435  return ADI_WDT_ERR_BAD_PRESCALE; /* invalid prescale value specified */
436  }
437 
438  return ADI_WDT_SUCCESS;
439 }

471 {
472 #ifdef ADI_DEBUG

474  if (ADI_WDT_INVALID_HANDLE(hDevice))

478  if (ADI_WDT_STATE_INITIALIZED != hDevice->InitState)
479  return ADI_WDT_ERR_NOT_INITIALIZED;

481

507  }
508  ADI_EXIT_CRITICAL_REGION();
509 
510  return ADI_WDT_SUCCESS;
