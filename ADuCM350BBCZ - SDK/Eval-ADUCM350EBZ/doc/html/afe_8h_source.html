196 /* Helper macro for adjusting MMR writes in a sequence, when modified from C code */
197 /* Parameters: */
198 /* - addr: MMR full address (32-bit), as defined in defADuCM350.h */
199 /* - data: value to be written to bits 24..0 in the MMR */
200 /* */
201 /* Details: */
202 /* - 0x80000000: sequencer command MSB set to "1", for an MMR write */
203 /* - addr[7:2] identifies the MMR for the sequencer, */
204 #define SEQ_MMR_WRITE(addr, data) (0x80000000 | ((addr & 0xFC) << 23) | data)
