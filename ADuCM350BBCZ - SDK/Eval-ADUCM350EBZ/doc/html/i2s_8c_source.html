268 ADI_I2S_RESULT_TYPE adi_I2S_Init(ADI_I2S_DEV_ID_TYPE const devID, ADI_I2S_DEV_HANDLE* const phDevice)
269 {
270  ADI_I2S_DEV_HANDLE hDevice;
271  ADI_I2S_RESULT_TYPE result;
272 
273 #ifdef ADI_DEBUG
274  if (devID >= ADI_I2S_MAX_DEVID)
275  return ADI_I2S_ERR_BAD_DEVICE_INDEX;
276 
277  if (ADI_I2S_STATE_UNINITIALIZED != pI2S0_DevData->initState)
278  return ADI_I2S_ERR_ALREADY_INITIALIZED;
279 #endif
280 
281  /* set local device handle */
282  hDevice = &I2S_DevData[devID];
283 
284  /* store a bad handle in case of failure */
285  *phDevice = NULL;
286 
287  /* do common init */
288  result = commonInit(hDevice);
289 
290  if (ADI_I2S_SUCCESS != result)
291  return result;
292 
293  ADI_I2S_CFG_TYPE const* pI2SCfg = &gI2SCfg[devID];
294 
295  /* Initialize the device based on the given configuration parameters */
296  hDevice->pI2S->I2S_MODE1 = pI2SCfg->I2S_MODE1;
297  hDevice->pI2S->I2S_MODE2 = pI2SCfg->I2S_MODE2;
298  hDevice->pI2S->I2S_CFG1 = pI2SCfg->I2S_CFG1;
299  hDevice->pI2S->I2S_CFG2 = pI2SCfg->I2S_CFG2;
300 
301  /* Set the audio format parameters based on the given configuration */
302  hDevice->b24bitMode = (pI2SCfg->I2S_MODE1 & BITM_I2S_I2S_MODE1_DATA_WIDTH) ? false : true;
303 
304 
305  /* Install the I2S interrupt handlers */
306  ADI_INSTALL_HANDLER(ADI_INT_I2S, I2S_Int_Handler);
307  ADI_INSTALL_HANDLER(ADI_INT_DMA_I2S, DMA_I2S_Int_Handler);
308 
309 #if (ADI_CFG_ENABLE_RTOS_SUPPORT == 1)
310  /* Create the semaphore for blocking mode support. */
311  if(adi_osal_SemCreateStatic(hDevice->SemMem, ADI_OSAL_MAX_SEM_SIZE_CHAR, &hDevice->hTxSem, 0) != ADI_OSAL_SUCCESS)
312  {
313  return ADI_I2S_ERR_SEMAPHORE_FAILED;
314  }
315 #endif /* ADI_CFG_ENABLE_RTOS_SUPPORT */
316 
317  /* init successful */
318  hDevice->initState = ADI_I2S_STATE_INITIALIZED;
319 
320  /* store the "device" handle last */
321  *phDevice = hDevice;
322 
323  return ADI_I2S_SUCCESS;
324 }

486 ADI_I2S_RESULT_TYPE adi_I2S_Transmit (ADI_I2S_DEV_HANDLE const hDevice,
487  uint8_t *pBuffer,
488  uint16_t const nBufferSize)
489 {
490 
491 #ifdef ADI_DEBUG
492  if (pI2S0_DevData != hDevice)
493  return ADI_I2S_ERR_BAD_DEVICE_HANDLE;
494 
495  if (ADI_I2S_STATE_UNINITIALIZED == hDevice->initState)
496  return ADI_I2S_ERR_NOT_INITIALIZED;
497 
498  if(NULL == pBuffer)
499  return ADI_I2S_ERR_NULL_DATA_POINTER;
500 
501  if(0 == nBufferSize)
502  return ADI_I2S_ERR_INVALID_DATA_SIZE;
503 
504 #endif
505  /* enable NVIC */
506  configureNVIC(hDevice, true);
507 
508  hDevice->pTxBuffer = pBuffer;
509  hDevice->TxRemaining = nBufferSize;
510  hDevice->result = ADI_I2S_SUCCESS;
511 
512 #if (ADI_I2S_CFG_ENABLE_DMA_SUPPORT == 1)
513  if (hDevice->bDmaMode)
514  {
515  hDevice->bDmaComplete = false;
516 
517  /* program initial descriptor pair */
518  hDevice->FunctionTable.pInitDescriptorsFcn(hDevice, I2S_CHANn);
519 
520  ADI_ENTER_CRITICAL_REGION();
521 
522  /* enable dma mode */
523  pADI_I2S->I2S_CFG1 |= (BITM_I2S_I2S_CFG1_AUTO_INCR | BITM_I2S_I2S_CFG1_DMA_EN);
524 
525  ADI_EXIT_CRITICAL_REGION();
526  }
527 #endif /* ADI_I2S_CFG_ENABLE_DMA_SUPPORT */
528 
529  /* In multi-threaded mode block using semaphore */
530  if(hDevice->bBlockingMode == true)
531  {
532 
533 #if (ADI_CFG_ENABLE_RTOS_SUPPORT == 1)
534 
535  /* In multi-threaed mode use semaphore to wait for the transfer complete. The semaphore is
536  posted from the ISR when the trasfer is complete */
537  if(adi_osal_SemPend(hDevice->hTxSem, ADI_OSAL_TIMEOUT_FOREVER) != ADI_OSAL_SUCCESS)
538  {
539  return ADI_I2S_ERR_SEMAPHORE_FAILED;
540  }
541 
542 #else /* Non multi-thread */
543 
544  if(hDevice->bDmaMode)
545  {
546  /* In DMA Mode wait for DMA complete flag to be set */
547  while(hDevice->bDmaComplete == false)
548  {
549  /* Instead of busy-waiting, we enter core sleep mode while waiting. Any
550  * interrupt that changes NumAvailable will also bring the processor
551  * out of core sleep and cause this call to return.
552  */
553 
554  SystemEnterLowPowerMode(ADI_SYS_MODE_CORE_SLEEP, /* wait in core sleep */
555  NULL,
556  0);
557  }
558  }
559  else
560  {
561  /* In non-DMA mode wait until the number of bytes remaining to transfer to become 0 */
562  while(hDevice->TxRemaining > 0)
563  {
564  /* Instead of busy-waiting, we enter core sleep mode while waiting. Any
565  * interrupt that changes NumAvailable will also bring the processor
566  * out of core sleep and cause this call to return.
567  */
568  SystemEnterLowPowerMode(ADI_SYS_MODE_CORE_SLEEP, /* wait in core sleep */
569  NULL,
570  0);
571  }
572  }
573 
574 #endif /* ADI_CFG_ENABLE_RTOS_SUPPORT */
575 
576  }
577  return ADI_I2S_SUCCESS;
578 
579 }

651  ADI_EXIT_CRITICAL_REGION();

1808 static ADI_I2S_RESULT_TYPE scheduleNextDescriptor(ADI_I2S_DEV_HANDLE const hDevice)
1809 {

1812  register uint32_t nextChunkSize;
1813 
1814  /* get the next descriptor type */
1815  pD->CCD = hDevice->nextDescrType;
1816

1818  hDevice->nextDescrType = (ADI_DMA_CCD_PRIMARY == hDevice->nextDescrType) ? ADI_DMA_CCD_ALTERNATE : ADI_DMA_CCD_PRIMARY;
1819

1821  pD->pSrcData = (void *)hDevice->pTxBuffer;
1822  pD->SrcInc = ADI_DMA_INCR_HALFWORD;
1823

1826  else

1828

1830

1832  So multiply by 2 to get the next transfer size. */

1838  hDevice->pTxBuffer += nextChunkSize;
1839  hDevice->TxRemaining -= nextChunkSize;
1840 
1841  }

1846  pD->DataLength = 0;
