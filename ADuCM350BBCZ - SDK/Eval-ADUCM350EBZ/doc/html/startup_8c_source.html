25 /*
26  Stack size considerations for RTOS environments:
27 
28  In the Cortex-M class processors, the hardware has two stacks: the main stack
29  and the process stack. The main stack is used at reset and by system exception
30  handlers. In simple applications (non-RTOS) you may only use this stack. In
31  more complex systems that use an RTOS you have a split between the main stack
32  which is used by the kernel and exception handlers and the process stack which
33  is used by the tasks and programmable interrupts (interrupts stack on top of
34  the currently active task, so tasks stacks should be allocated for worst-case
35  interrupt scenarios). In the uCOS case, the process stack is the one that gets
36  context-switched and used by the tasks, where we store the registers, etc.
37 
38  As RTOS tasks specify their own stack area, a minimal size of 4 for the process
39  stack in order to retain symbols, etc., seems OK and we would not need any more.
40  The main stack needs to be big enough to initialize the RTOS, run interrupts etc.
41 
42  As always, stack and heap size optimizations is a balancing act between usage
43  and available memory. It is entirely application-dependant.
44 
45 */

56 //****************************************************************************************************************
57 //
58 // When operating in an RTOS context, interrupts are dispatched by the NVIC controller
59 // to a second level dispatch function known as the OSAL (operating system abstraction layer) dispatch.
60 // The OSAL dispatch has common code that will interact with the RTOS.
61 //
62 // OSAL dispatched interrupts must pass their Interrupt ID (iid) to the common OSAL dispatcher.
63 //
64 // System interrupts (Reset through SysTick) should never dispatch through the OSAL.
65 //
66 // The IntFunc typedef is used for System Interrupts handlers. The parameter list for these interrupts is always 'void'.
67 //
68 // The ADIIntFunc typdef is used for Driver Interrupt Handlers. The parameter list for these handlers
69 // will change depending on the RTOS context. In an RTOS context the IID (Interrupt ID) must be passed to the OSAL
70 // dispatcher (along with a second arg of type void * which is handler specific). In a non-RTOS context the handler
71 // parameters is simply 'void'.
72 //
73 //*****************************************************************************************************************

177 #ifdef RELOCATE_IVT
178 /*
179  A relocated IVT is requested. Provision for IVT relocation
180  to RAM during startup. This allows for dynamic interrupt
181  vector patching required by RTOS. Places the relocated IVT
182  at the start of RAM. Note: the IVT placement is required
183  to be next power-of-two of the vector table size. So the
184  IVT includes 61 programmable interrupts, 15 system exception
185  vectors and the main stack pointer, therefore we need
186  (61 + 15 + 1)*4 = 308 bytes, which rounds up to a 512 (0x200)
187  address boundary (which address 0x20000000 satisfies).
188 
189  Also note use of the "__no_init" attribute to force the
190  relocated IVT into the ".noinit" section. This prevents
191  the CRTL startup sequence from initializing the relocated
192  IVT after we have activated it during the reset vector.
193  so that the CRTL does not clear it.
194 */
195 #define RELOCATION_ADDRESS (0x20000000)
196 #define RELOCATION_ALIGNMENT (0x200)
197 #define SIZEOF_IVT sizeof(__vector_table)
198 #if defined ( __ICCARM__ )
199  #pragma data_alignment=RELOCATION_ALIGNMENT /* IAR */
200 #elif defined (__CC_ARM)
201  __align(RELOCATION_ALIGNMENT) /* Keil */
202 #else
203  #pragma message("WARNING: NO ALIGHMENT DEFINED FOR IVT RELOCATION")
204 #endif
205 /* reserve no-init aligned IVT space at top of RAM */
206 SECTION_PLACE(KEEP_VAR(__no_init IntVector __relocated_vector_table[SIZEOF_IVT/4]), RELOCATION_ADDRESS);
207 #endif

243 //*****************************************************************************
244 // Function : ResetISR (-15)
245 // Description : Reset event handler
246 //*****************************************************************************
247 
248 WEAK_FUNC (void ResetISR (void)) {
249 
250 
251  /* reset the main SP to clean up any leftovers from the boot rom. */
252  __set_MSP( (uint32_t) __vector_table[0].__ptr );
253 
254  /* Unlock the PWRMOD register by writing the two keys to the PWRKEY register */
255  pADI_PWR->PWRKEY = PWRKEY_VALUE_KEY1;
256  pADI_PWR->PWRKEY = PWRKEY_VALUE_KEY2;
257 
258  /* set the RAM0_RET bit so the entire 16k of SRAM Bank0 is hibernate-preserved */
259  pADI_PWR->PWRMOD |= (1 << BITP_PWR_PWRMOD_RAM0_RET);
260 
261 #ifdef RELOCATE_IVT
262  int i;
263  uint8_t *pSrc, *pDst;
264 
265  /* copy the IVT (avoid use of memcpy here so it does not become locked into flash) */
266  for (i=0, pSrc=(uint8_t*)__vector_table, pDst=(uint8_t*)__relocated_vector_table; i<SIZEOF_IVT; i++)
267  *pDst++ = *pSrc++;
268 #endif
269 
270 #ifdef __GNUC__
271 
272  unsigned long *pulSrc, *pulDest;
273 
274  // Copy initialised data from flash into RAM
275  pulSrc = &_etext;
276  for(pulDest = &_data; pulDest < &_edata; )
277  {
278  *pulDest++ = *pulSrc++;
279  }
280 
281  // Clear the bss segment
282  for(pulDest = &_bss; pulDest < &_ebss; )
283  {
284  *pulDest++ = 0;
285  }
286  // Call application main directly.
287  main();
288 
289 #elif __ICCARM__
290 
291  // Call IAR system startup
292  __iar_init_core();
293  __iar_init_vfp();
294  __cmain();
295 
296 #else
297 
298  // Call application main directly.
299  main();
300 
301 #endif
302 
303  // Stick here if main returns
304  while(1);
305 }
