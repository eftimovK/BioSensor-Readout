216 void SystemInit (void)
217 {
218 
219 /* version check */
220 #if (ADI_VERSION_CURRENT < ADI_VERSION_2_1_0_0)
221 #error "Unsupported version."
222 #endif
223 
224  /* Switch the Interrupt Vector Table Offset Register
225  * (VTOR) to point to the relocated IVT in SRAM
226  */
227 
228  ADI_ENTER_CRITICAL_REGION(); // do all this in safe way
229 
230  // switch from boot ROM IVT to application's IVT
231  // set the System Control Block, Vector Table Offset Register
232 #ifdef RELOCATE_IVT
233  SCB->VTOR = (uint32_t) &__relocated_vector_table;
234 #else
235  SCB->VTOR = (uint32_t) &__vector_table;
236 #endif
237 
238  // set all three (USGFAULTENA, BUSFAULTENA, and MEMFAULTENA) fault enable bits
239  // in the System Control Block, System Handler Control and State Register
240  // otherwise these faults are handled as hard faults
241  SCB->SHCSR = SCB_SHCSR_USGFAULTENA_Msk |
242  SCB_SHCSR_BUSFAULTENA_Msk |
243  SCB_SHCSR_MEMFAULTENA_Msk ;
244 
245  // flush instruction and data pipelines to insure assertion of new settings
246  __ISB(); // MUST OCCURE IMMEDIATELY AFTER UPDATING SCB->CPACR!!!
247  __DSB();
248 
249  ADI_EXIT_CRITICAL_REGION();
250 
251  /* restore clocks to a known power-up reset condition */
252 
253  /*
254  Reset sequence should be...
255  1) OSCCTRL. Set bit 1 HFOSCEN. This will bring up the internal oscillator.
256  2) CLKCON0. Clear bits 1:0 to 00. This will bring in the internal oscillator as the
257  system clock source. Note, you do not need a wait time for steps 1 or 2.
258  3) Simply step through OSCCTRL and CLKCON0,1,3,4,5 in order and set these to their
259  reset states.
260 
261  NOTE for soft-reset via NVIC:
262  Before issuing a SYSRESETREQ, allowing the Cortex-M3 to assert a reset request signal to
263  the system’s reset generator, the HFOSCEN (OSCCTRL bit 1) must be set. This will ensure
264  that all system componenets will be reset properly. This is independent of Root Clock Mux
265  and SPLL Clock Mux settings.
266 
267  */
268 
269  /* Enable internal HF & LF oscillators */
270  pADI_PWR->OSCKEY = OSCKEY_UNLOCK;
271  pADI_PWR->OSCCTRL |= ((1 << BITP_PWR_OSCCTRL_HFOSCEN) | (1 << BITP_PWR_OSCCTRL_LFOSCEN));
272 
273  /* wait for HF OSC to stabilize */
274  while ( !(pADI_PWR->OSCCTRL & (1 << BITP_PWR_OSCCTRL_HFOSCOK)) );
275 
276  /* Switch over to the internal HF oscillator */
277  pADI_SYSCLK->CLKCON0 &= ~(BITM_SYSCLK_CLKCON0_CLKMUX);
278 
279  /* complete remaining reset sequence */
280  pADI_SYSCLK->CLKCON0 = CLKCON0_RVAL;
281  pADI_SYSCLK->CLKCON1 = CLKCON1_RVAL;
282  /* no CLKCON2 */
283  pADI_SYSCLK->CLKCON3 = CLKCON3_RVAL;
284  pADI_SYSCLK->CLKCON4 = CLKCON4_RVAL;
285  pADI_SYSCLK->CLKCON5 = CLKCON5_RVAL;
286 
287  /* Change the PLL input source from the default HFOSC to HFXTAL */
288  /* to satisfy the majority of use cases. It has no effect if */
289  /* no PLL is used. */
290  SetSystemClockMux(ADI_SYS_CLOCK_MUX_SPLL_HF_XTAL);
291 
292  /* disable external HF crystal oscillator */
293  /* (don't disable LF crystal or the RTC will lose time */
294  pADI_PWR->OSCKEY = OSCKEY_UNLOCK;
295  pADI_PWR->OSCCTRL &= ~BITM_PWR_OSCCTRL_HFXTALEN;
296 
297  /* compute new internal clocks based on the newly reset controller */
298  SystemCoreClockUpdate();
299 }

310 void SystemCoreClockUpdate (void)
311 {
312  uint32_t val;
313  uint32_t tmp;
314 
315  /* Update Core Clock sources */
316 
317 #ifdef ADI_DEBUG
318  /* "lfclock" is only used during debug checks... */
319  /* LF clock is always 32k, whether osc or xtal */
320  lfClock = __LFCLK; /* for beep, wdt and lcd */
321 #endif
322 
323  /* update the HF clock */
324  switch (pADI_SYSCLK->CLKCON0 & BITM_SYSCLK_CLKCON0_CLKMUX ) {
325 
326  case HFMUX_INTERNAL_OSC_VAL:
327  hfClock = __HFOSC;
328  break;
329 
330  case HFMUX_EXTERNAL_XTAL_VAL:
331  hfClock = __HFXTAL;
332  break;
333 
334  case HFMUX_SYSTEM_PLL_VAL:
335  /* Calculate System PLL output frequency */
336  if (pADI_SYSCLK->CLKCON0 & BITM_SYSCLK_CLKCON0_PLLMUX) {
337  /* PLL input from HFXTAL */
338  val = __HFXTAL;
339  } else {
340  /* PLL input from HFOSC */
341  val = __HFOSC;
342  }
343 
344  /* PLL N multiplier */
345  tmp = (pADI_SYSCLK->CLKCON3 & BITM_SYSCLK_CLKCON3_SPLLNSEL) >> BITP_SYSCLK_CLKCON3_SPLLNSEL;
346  val = val * tmp;
347 
348  /* PLL M divider */
349  tmp = pADI_SYSCLK->CLKCON3 & BITM_SYSCLK_CLKCON3_SPLLMSEL;
350  /* Adjustment needs to be made if M is 2 or 4, if M is 1 leaves it as is */
351  switch (tmp) {
352  case ADI_SYS_CLOCK_PLL_MSEL_4:
353  /* Divide by 4 */
354  val = val >> 2;
355  break;
356  case ADI_SYS_CLOCK_PLL_MSEL_2:
357  /* Divide by 2 */
358  val = val >> 1;
359  break;
360  }
361 
362  /* PLL divide by 2 */
363  if (pADI_SYSCLK->CLKCON3 & BITM_SYSCLK_CLKCON3_SPLLDIV2) {
364  val = val >> 1;
365  }
366 
367  hfClock = val;
368  break;
369 
370  case HFMUX_GPIO_VAL:
371  hfClock = gpioClock;
372  break;
373 
374  } /* end switch */
375 }

540 ADI_SYS_CLOCK_MUX_ID GetSystemClockMux(ADI_SYS_CLOCK_MUX_GROUP_ID id) {
541  uint16_t mask;
542  uint16_t position;
543  uint16_t offset;
544  uint16_t tmp;
545 
546  /* The following logic depends on the enumeration order being correct */
547  switch (id) {
548 
549  /* SPLL/UPLL clock source */
550  case ADI_SYS_CLOCK_MUX_SPLL:
551  mask = BITM_SYSCLK_CLKCON0_PLLMUX;
552  position = BITP_SYSCLK_CLKCON0_PLLMUX;
553  offset = ADI_SYS_CLOCK_MUX_SPLL_HF_OSC;
554  break;
555 
556  /* lf_clk source */
557  case ADI_SYS_CLOCK_MUX_LFCLK:
558  mask = BITM_SYSCLK_CLKCON0_LFCLKMUX;
559  position = BITP_SYSCLK_CLKCON0_LFCLKMUX;
560  offset = ADI_SYS_CLOCK_MUX_LFCLK_LFOSC;
561  break;
562 
563  /* Clock output clock source */
564  case ADI_SYS_CLOCK_MUX_OUTPUT:
565  mask = BITM_SYSCLK_CLKCON0_CLKCOUT;
566  position = BITP_SYSCLK_CLKCON0_CLKCOUT;
567  offset = ADI_SYS_CLOCK_MUX_OUTPUT_ROOT_CLK;
568  break;
569 
570  /* root_clk source */
571  case ADI_SYS_CLOCK_MUX_ROOT:
572  mask = BITM_SYSCLK_CLKCON0_CLKMUX;
573  position = BITP_SYSCLK_CLKCON0_CLKMUX;
574  offset = ADI_SYS_CLOCK_MUX_ROOT_HFOSC;
575  break;
576 
577  } /* end switch */
578 
579  tmp = (pADI_SYSCLK->CLKCON0 & mask) >> position;
580  tmp += offset;
581 
582  return (ADI_SYS_CLOCK_MUX_ID)tmp;
583

618 uint32_t SystemGetClockFrequency (ADI_SYS_CLOCK_ID id)
619 {
620 
621  uint32_t src, div;
622 
623 #ifdef ADI_DEBUG
624  // trap here if the app fails to call SystemInit().
625  if (0 == hfClock || 0 == lfClock)
626  while(1); // MUST CALL SystemInit() to get valid clock internals: FIX APP CODE!!!
627 #endif
628 
629  /* refresh internal clock variables */
630  SystemCoreClockUpdate();
631 
632  switch (id) {
633 
634  /* USB domain */
635  case ADI_SYS_CLOCK_USBPHY:
636  case ADI_SYS_CLOCK_USBCTL:
637  src = __USBPHYCLK;
638 
639  if (1 == ((pADI_SYSCLK->CLKCON1 & BITM_SYSCLK_CLKCON1_HCLKDIVCNT) >> BITP_SYSCLK_CLKCON1_HCLKDIVCNT) ) {
640 
641  /* if HCLKDIVCNT == 1, divide is 1 */
642  div = 1;
643 
644  } else {
645 
646  /* if HCLKDIVCNT != 1, divide is based on USBCTLCLKDIVMUX */
647 
648  /* start with HCLKDIVCNT... */
649  div = (pADI_SYSCLK->CLKCON1 & BITM_SYSCLK_CLKCON1_HCLKDIVCNT) >> BITP_SYSCLK_CLKCON1_HCLKDIVCNT;
650 
651  if (pADI_SYSCLK->CLKCON1 & BITM_SYSCLK_CLKCON1_USBCTLCLKDIVMUX) {
652 
653  /* if USBCTLCLKDIVMUX bit is set, divider is 1/2 HCLKDIVCNT */
654  div >>= 2;
655  }
656  }
657  break;
658 
659  /* PCLOCK domain */
660  case ADI_SYS_CLOCK_PCLK:
661  case ADI_SYS_CLOCK_I2C:
662  case ADI_SYS_CLOCK_UART:
663  case ADI_SYS_CLOCK_I2S:
664  case ADI_SYS_CLOCK_SPI0:
665  case ADI_SYS_CLOCK_SPI1:
666  src = hfClock;
667  div = (pADI_SYSCLK->CLKCON1 & BITM_SYSCLK_CLKCON1_PCLKDIVCNT) >> BITP_SYSCLK_CLKCON1_PCLKDIVCNT;
668  break;
669 
670  /* HCLOCK & ACLOCK domain */
671  case ADI_SYS_CLOCK_SPIH:
672  case ADI_SYS_CLOCK_CORE:

990 ADI_SYS_RESULT_TYPE SystemClockSourcePowerUp (ADI_SYS_CLOCK_SOURCE_ID id, ADI_SYS_CLOCK_TIMEOUT_TYPE timeout) {
991  ADI_SYS_RESULT_TYPE result = ADI_SYS_SUCCESS;
992  uint32_t timeoutVal;
993  uint16_t nsel;
994  uint16_t regVal;
995 
996  /* If it's already up and stable, do nothing */
997  if (GetClockSourceStatus(id) != ADI_SYS_CLOCK_SOURCE_ENABLED_STABLE) {
998 
999 #if (UPLL_LOCK_WORKAROUND == 1)
1000 
1001  if (id == ADI_SYS_CLOCK_SOURCE_UPLL) {
1002 
1003  /* The following is split into 2 critical regions, because of the long timeout. */
1004 
1005  /* critical region */
1006  ADI_ENTER_CRITICAL_REGION();
1007 
1008  /* Save the programmed UPLL NSEL setting. */
1009  /* CLKCON4 should always have UPLLMSEL=0x2 and UPLLNSEL=0x0F for 60MHz output. */
1010  /* Writing it as a general function. */
1011  regVal = pADI_SYSCLK->CLKCON4;
1012 
1013  nsel = (regVal & BITM_SYSCLK_CLKCON4_UPLLNSEL) >> BITP_SYSCLK_CLKCON4_UPLLNSEL;
1014 
1015  /* Overwrite the UPLL settings using UPLLNSEL=0x3F to precharge the loop filter, */
1016  /* and enable UPLL. */
1017  regVal &= ~BITM_SYSCLK_CLKCON4_UPLLNSEL;
1018  regVal |= (0x3F << BITP_SYSCLK_CLKCON4_UPLLNSEL);
1019  regVal |= (1 << BITP_SYSCLK_CLKCON4_UPLLEN);
1020  pADI_SYSCLK->CLKCON4 = regVal;
1021 
1022  /* end critical region */
1023  ADI_EXIT_CRITICAL_REGION();
1024 
1025  /* Approximate scaling of the timeout value with HCLK frequency */
1026  timeoutVal = scaleTimeout(UPLL_LOCK_WORKAROUND_PRECHARGE_TIME);
1027 
1028  /* Wait while the loop filter is precharging */
1029  for (; timeoutVal; timeoutVal--)
1030  ;
1031 
1032  /* Now restore the original UPLL NSEL setting. */
1033 
1034  /* critical region */
1035  ADI_ENTER_CRITICAL_REGION();
1036 
1037  /* regVal should not have changed from the previous value */
1038  /* However it's possible (and probably wrong) to be modified */
1039  /* for example in an interrupt. */
1040  regVal = pADI_SYSCLK->CLKCON4;
1041  regVal &= ~BITM_SYSCLK_CLKCON4_UPLLNSEL;
1042  regVal |= (nsel << BITP_SYSCLK_CLKCON4_UPLLNSEL);
1043  pADI_SYSCLK->CLKCON4 = regVal;
1044 
1045  /* end critical region */
1046  ADI_EXIT_CRITICAL_REGION();
1047  }
1048  else {
1049  SystemEnableClockSource(id, true);
1050  }
1051 #else
1052  SystemEnableClockSource(id, true);
1053 #endif
1054 
1055  if (timeout != ADI_SYS_CLOCK_TIMEOUT_DONT_WAIT_STABLE) {
1056  /* Timeout required */
1057  if (timeout == ADI_SYS_CLOCK_TIMEOUT_USE_DEFAULT) {
1058  /* Use the timeout defaults */
1059  timeoutVal = CLOCK_POWERUP_TIMEOUT;
1060  } else {
1061  /* Timeout value explicitely passed on by the user */
1062  timeoutVal = timeout;
1063  }
1064 
1065  /* Approximate scaling of the timeout value with HCLK frequency */
1066  timeoutVal = scaleTimeout(timeoutVal);
1067 
1068  for ( ; timeoutVal && (GetClockSourceStatus(id) != ADI_SYS_CLOCK_SOURCE_ENABLED_STABLE); timeoutVal--)
1069  ;
1070 
1071  if (!timeoutVal) {
1072  result = ADI_SYS_ERR_CLOCK_SOURCE_TIMEOUT;
1073  }
1074  }
1075  }
1076 
1077  return result;
1078 }

1545  bool_t volatile * pbInterruptOccurred,
1546  const uint8_t PriorityMask)
1547 {

1549  uint16_t scrSetBits = 0u;
1550  uint16_t scrClrBits = 0u;
1551 
1552 #ifdef ADI_DEBUG
1553 
1554  /* verify the requested priority mask bits are right-justified and don't exceed __NVIC_PRIO_BITS in width */
1555  if ( PriorityMask & ~((1 << __NVIC_PRIO_BITS) - 1) )
1556  return ADI_SYS_ERR_PARAM_OUT_OF_RANGE;
1557 
1558 #endif /* ADI_DEBUG */

1561  if(NULL == pbInterruptOccurred) {
1562  scrSetBits |= SLEEPONEXIT_BIT;
1563 
1564  /* point to private control variable when in hardware mode */
1565  pbInterruptOccurred = &privateIntOccFlag;
1566  }
1567 
1568  /* pre-calculate the deepsleep set/clear bits */
1569  switch (PowerMode) {
1570  case ADI_SYS_MODE_ACTIVE:
1571  return ADI_SYS_SUCCESS;
1572 
1573  case ADI_SYS_MODE_CORE_SLEEP:
1574  case ADI_SYS_MODE_SYS_SLEEP:
1575  scrClrBits |= SLEEPDEEP_BIT;
1576  break;
1577 
1578  case ADI_SYS_MODE_HIBERNATE:
1579  scrSetBits |= SLEEPDEEP_BIT;

1583  default:
1584  return ADI_SYS_ERR_INVALID_POWER_MODE;
1585 #endif /*ADI_DEBUG*/
1586 
1587  } /* end switch */
1588

1590  ADI_ENTER_CRITICAL_REGION();

1592  { /* these three lines must be in a success-checking loop if they are not inside critical section */

1595  pADI_PWR->PWRKEY = PWRKEY_VALUE_KEY2;

1599  }

1602  SCB->SCR = ((SCB->SCR | scrSetBits) & ~scrClrBits);

1608  __set_BASEPRI(PriorityMask << (8-__NVIC_PRIO_BITS));

1624  * allows *pbInterruptOccurred to be set during interrupt handling.

1628  ADI_ENTER_CRITICAL_REGION();
1629

1637  __set_BASEPRI(savedPriority);
